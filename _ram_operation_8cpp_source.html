56 /** get indexable element */
57 std::unique_ptr<RamValue> getIndexElement(RamCondition* c, size_t& element, size_t level) {
58  if (auto* binRelOp = dynamic_cast<RamBinaryRelation*>(c)) {
59  if (binRelOp->getOperator() == BinaryConstraintOp::EQ) {
60  if (auto* lhs = dynamic_cast<RamElementAccess*>(binRelOp->getLHS())) {
61  RamValue* rhs = binRelOp->getRHS();
62  if (lhs->getLevel() == level && (rhs->isConstant() || rhs->getLevel() < level)) {
63  element = lhs->getElement();
64  return binRelOp->takeRHS();
65  }
66  }
67  if (auto* rhs = dynamic_cast<RamElementAccess*>(binRelOp->getRHS())) {
68  RamValue* lhs = binRelOp->getLHS();
69  if (rhs->getLevel() == level && (lhs->isConstant() || lhs->getLevel() < level)) {
70  element = rhs->getElement();
71  return binRelOp->takeLHS();
72  }
73  }
74  }
75  }
76  return std::unique_ptr<RamValue>(nullptr);
77 }

80 /** add condition */
81 void RamScan::addCondition(std::unique_ptr<RamCondition> c, const RamOperation& root) {
82  // use condition to narrow scan if possible
83  if (c->getLevel() == level) {
84  size_t element = 0;
85  if (std::unique_ptr<RamValue> value = getIndexElement(c.get(), element, level)) {
86  keys |= (1 << element);
87  if (queryPattern[element] == nullptr) {
88  queryPattern[element] = std::move(value);
89  } else {
90  std::unique_ptr<RamValue> field = std::make_unique<RamElementAccess>(level, element);
91  RamSearch::addCondition(std::make_unique<RamBinaryRelation>(
92  BinaryConstraintOp::EQ, std::move(field), std::move(value)),
93  root);
94  }
95  return;
96  }
97  }
98 
99  // otherwise: use default handling
100  RamSearch::addCondition(std::move(c), root);
101 }

176 /** add condition */
177 void RamAggregate::addCondition(std::unique_ptr<RamCondition> c, const RamOperation& root) {
178  // use condition to narrow scan if possible
179  if (c->getLevel() == level) {
180  size_t element = 0;
181  if (std::unique_ptr<RamValue> value = getIndexElement(c.get(), element, level)) {
182  if (element > 0 || relation->getName().find("__agg") == std::string::npos) {
183  keys |= (1 << element);
184  if (pattern[element] == nullptr) {
185  pattern[element] = std::move(value);
186  } else {
187  std::unique_ptr<RamValue> field(new RamElementAccess(level, element));
188  RamSearch::addCondition(std::make_unique<RamBinaryRelation>(BinaryConstraintOp::EQ,
189  std::move(field), std::move(value)),
190  root);
191  }
192  } else {
193  std::unique_ptr<RamValue> field(new RamElementAccess(level, element));
194  std::unique_ptr<RamCondition> eq(
195  new RamBinaryRelation(BinaryConstraintOp::EQ, std::move(field), std::move(value)));
196  if (condition != nullptr) {
197  condition = std::make_unique<RamAnd>(std::move(condition), std::move(eq));
198  } else {
199  condition.swap(eq);
200  }
201  }
202  return;
203  }
204  }
205 
206  // otherwise: use default handling
207  RamSearch::addCondition(std::move(c), root);
208 }
