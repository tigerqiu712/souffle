67 /**
68  * A base class for AstConstraintAnalysis collecting constraints for an analysis
69  * by visiting every node of a given AST. The collected constraints are
70  * then utilized to obtain the desired analysis result.
71  *
72  * @tparam AnalysisVar the type of variable (and included property space)
73  * to be utilized by this analysis.
74  */
75 template <typename AnalysisVar>
76 class AstConstraintAnalysis : public AstVisitor<void> {
77  using value_type = typename AnalysisVar::property_space::value_type;
78 
79  /** The list of constraints making underlying this analysis */
80  Problem<AnalysisVar> constraints;
81 
82  /** A map mapping variables to unique instances to facilitate the unification of variables */
83  std::map<std::string, AnalysisVar> variables;
84 
85 protected:
86  // a few type definitions
87  using constraint_type = std::shared_ptr<Constraint<AnalysisVar>>;
88  using solution_type = std::map<const AstArgument*, value_type>;
89 
90  /**
91  * A utility function mapping an AstArgument to its associated analysis variable.
92  *
93  * @param arg the AST argument to be mapped
94  * @return the analysis variable representing its associated value
95  */
96  AnalysisVar getVar(const AstArgument& arg) {
97  const auto* var = dynamic_cast<const AstVariable*>(&arg);
98  if (!var) {
99  // no mapping required
100  return AnalysisVar(arg);
101  }
102 
103  // filter through map => always take the same variable
104  auto res = variables.insert(std::make_pair(var->getName(), AnalysisVar(var)));
105  return res.first->second;
106  }
107 
108  /**
109  * A utility function mapping an AstArgument to its associated analysis variable.
110  *
111  * @param arg the AST argument to be mapped
112  * @return the analysis variable representing its associated value
113  */
114  AnalysisVar getVar(const AstArgument* arg) {
115  return getVar(*arg);
116  }
117 
118  /** Adds another constraint to the internally maintained list of constraints */
119  void addConstraint(const constraint_type& constraint) {
120  constraints.add(constraint);
121  }
122 
123 public:
124  /**
125  * Runs this constraint analysis on the given clause.
126  *
127  * @param clause the close to be analysed
128  * @param debug a flag enabling the printing of debug information
129  * @return an assignment mapping a property to each argument in the given clause
130  */
131  solution_type analyse(const AstClause& clause, bool debug = false) {
132  // collect constraints
133  visitDepthFirstPreOrder(clause, *this);
134 
135  // solve constraints
136  auto ass = constraints.solve();
137 
138  // print debug information if desired
139  if (debug) {
140  std::cout << "Clause: " << clause << "\n";
141  std::cout << "Problem:\n" << constraints << "\n";
142  std::cout << "Solution:\n" << ass << "\n";
143  }
144 
145  // convert assignment to result
146  solution_type res;
147  visitDepthFirst(clause, [&](const AstArgument& cur) { res[&cur] = ass[getVar(cur)]; });
148  return res;
149  }
150 };

339  // define analysis ..
340  struct Analysis : public AstConstraintAnalysis<BoolDisjunctVar> {
341  std::set<const AstAtom*> ignore;
342 
343  // #1 - atoms are producing grounded variables
344  void visitAtom(const AstAtom& cur) override {
345  // some atoms need to be skipped (head or negation)
346  if (ignore.find(&cur) != ignore.end()) {
347  return;
348  }
349 
350  // all arguments are grounded
351  for (const auto& arg : cur.getArguments()) {
352  addConstraint(isTrue(getVar(arg)));
353  }
354  }
355 
356  // #2 - negations need to be skipped
357  void visitNegation(const AstNegation& cur) override {
358  // add nested atom to black-list
359  ignore.insert(cur.getAtom());
360  }
361 
362  // #3 - also skip head
363  void visitClause(const AstClause& clause) override {
364  // ignore head
365  ignore.insert(clause.getHead());
366  }
367 
368  // #4 - binary equality relations propagates groundness
369  void visitBinaryConstraint(const AstBinaryConstraint& cur) override {
370  // only target equality
371  if (cur.getOperator() != BinaryConstraintOp::EQ) {
372  return;
373  }
374 
375  // if equal, link right and left side
376  auto lhs = getVar(cur.getLHS());
377  auto rhs = getVar(cur.getRHS());
378 
379  addConstraint(imply(lhs, rhs));
380  addConstraint(imply(rhs, lhs));
381  }
382 
383  // #5 - record init nodes
384  void visitRecordInit(const AstRecordInit& init) override {
385  auto cur = getVar(init);
386 
387  std::vector<BoolDisjunctVar> vars;
388 
389  // if record is grounded, so are all its arguments
390  for (const auto& arg : init.getArguments()) {
391  auto arg_var = getVar(arg);
392  addConstraint(imply(cur, arg_var));
393  vars.push_back(arg_var);
394  }
395 
396  // if all arguments are grounded, so is the record
397  addConstraint(imply(vars, cur));
398  }
399 
400  // #6 - constants are also sources of grounded values
401  void visitConstant(const AstConstant& c) override {
402  addConstraint(isTrue(getVar(c)));
403  }
404 
405  // #7 - aggregators are grounding values
406  void visitAggregator(const AstAggregator& c) override {
407  addConstraint(isTrue(getVar(c)));
408  }
409 
410  // #8 - functors with grounded values are grounded values
411  void visitUnaryFunctor(const AstUnaryFunctor& cur) override {
412  auto fun = getVar(cur);
413  auto arg = getVar(cur.getOperand());
414 
415  addConstraint(imply(arg, fun));
416  }
417 
418  void visitBinaryFunctor(const AstBinaryFunctor& cur) override {
419  auto fun = getVar(cur);
420  auto lhs = getVar(cur.getLHS());
421  auto rhs = getVar(cur.getRHS());
422 
423  addConstraint(imply({lhs, rhs}, fun));
424  }

426  void visitTernaryFunctor(const AstTernaryFunctor& cur) override {

429  auto a1 = getVar(cur.getArg(1));
430  auto a2 = getVar(cur.getArg(2));
431 
432  addConstraint(imply({a0, a1, a2}, fun));

434  };

437  return Analysis().analyse(clause);

640 TypeConstraint isSubtypeOfComponent(const TypeVar& a, const TypeVar& b, int index) {
641  struct C : public Constraint<TypeVar> {
642  TypeVar a;
643  TypeVar b;
644  unsigned index;
645 
646  C(TypeVar a, TypeVar b, int index) : a(std::move(a)), b(std::move(b)), index(index) {}
647 
648  bool update(Assignment<TypeVar>& ass) const override {
649  // get list of types for b
650  const TypeSet& recs = ass[b];
651 
652  // if it is (not yet) constrainted => skip
653  if (recs.isAll()) {
654  return false;
655  }
656 
657  // compute new types for a and b
658  TypeSet typesA;
659  TypeSet typesB;
660 
661  // iterate through types of b
662  for (const Type& t : recs) {
663  // only retain records
664  if (!isRecordType(t)) {
665  continue;
666  }
667  const auto& rec = static_cast<const RecordType&>(t);
668 
669  // of proper size
670  if (rec.getFields().size() <= index) {
671  continue;
672  }
673 
674  // this is a valid type for b
675  typesB.insert(rec);
676 
677  // and its corresponding field for a
678  typesA.insert(rec.getFields()[index].type);
679  }
680 
681  // combine with current types assigned to a
682  typesA = getGreatestCommonSubtypes(ass[a], typesA);
683 
684  // update values
685  bool changed = false;
686  if (recs != typesB) {
687  ass[b] = typesB;
688  changed = true;
689  }
690 
691  if (ass[a] != typesA) {
692  ass[a] = typesA;
693  changed = true;
694  }
695 
696  // done
697  return changed;
698  }
699 
700  void print(std::ostream& out) const override {
701  out << a << " <: " << b << "::" << index;
702  }
703  };
704 
705  return std::make_shared<C>(a, b, index);
706 }

713 /**
714  * A utility function utilized by the finishParsing member function to update a type environment
715  * out of a given list of types in the AST
716  *
717  * @param types the types specified in the input file, contained in the AST
718  * @param env the type environment to be updated
719  */
720 void TypeEnvironmentAnalysis::updateTypeEnvironment(const AstProgram& program) {
721  // build up new type system based on defined types
722 
723  // create all type symbols in a first step
724  for (const auto& cur : program.getTypes()) {
725  // support faulty codes with multiple definitions
726  if (env.isType(cur->getName())) {
727  continue;
728  }
729 
730  // create type within type environment
731  if (auto* t = dynamic_cast<const AstPrimitiveType*>(cur)) {
732  if (t->isNumeric()) {
733  env.createNumericType(cur->getName());
734  } else {
735  env.createSymbolType(cur->getName());
736  }
737  } else if (dynamic_cast<const AstUnionType*>(cur)) {
738  // initialize the union
739  env.createUnionType(cur->getName());
740  } else if (dynamic_cast<const AstRecordType*>(cur)) {
741  // initialize the record
742  env.createRecordType(cur->getName());
743  } else {
744  std::cout << "Unsupported type construct: " << typeid(cur).name() << "\n";
745  assert(false && "Unsupported Type Construct!");
746  }
747  }
748 
749  // link symbols in a second step
750  for (const auto& cur : program.getTypes()) {
751  Type* type = env.getModifiableType(cur->getName());
752  assert(type && "It should be there!");
753 
754  if (dynamic_cast<const AstPrimitiveType*>(cur)) {
755  // nothing to do here
756  } else if (auto* t = dynamic_cast<const AstUnionType*>(cur)) {
757  // get type as union type
758  auto* ut = dynamic_cast<UnionType*>(type);
759  if (!ut) {
760  continue; // support faulty input
761  }
762 
763  // add element types
764  for (const auto& cur : t->getTypes()) {
765  if (env.isType(cur)) {
766  ut->add(env.getType(cur));
767  }
768  }
769  } else if (auto* t = dynamic_cast<const AstRecordType*>(cur)) {
770  // get type as record type
771  auto* rt = dynamic_cast<RecordType*>(type);
772  if (!rt) {
773  continue; // support faulty input
774  }
775 
776  // add fields
777  for (const auto& f : t->getFields()) {
778  if (env.isType(f.type)) {
779  rt->add(f.name, env.getType(f.type));
780  }
781  }
782  } else {
783  std::cout << "Unsupported type construct: " << typeid(cur).name() << "\n";
784  assert(false && "Unsupported Type Construct!");
785  }
786  }
787 }

789 /* Return a new clause with type-annotated variables */
790 AstClause* createAnnotatedClause(
791  const AstClause* clause, const std::map<const AstArgument*, TypeSet> argumentTypes) {
792  // Annotates each variable with its type based on a given type analysis result
793  struct TypeAnnotator : public AstNodeMapper {
794  const std::map<const AstArgument*, TypeSet>& types;
795 
796  TypeAnnotator(const std::map<const AstArgument*, TypeSet>& types) : types(types) {}
797 
798  std::unique_ptr<AstNode> operator()(std::unique_ptr<AstNode> node) const override {
799  if (auto* var = dynamic_cast<AstVariable*>(node.get())) {
800  std::stringstream newVarName;
801  newVarName << var->getName() << "&isin;" << types.find(var)->second;
802  return std::make_unique<AstVariable>(newVarName.str());
803  } else if (auto* var = dynamic_cast<AstUnnamedVariable*>(node.get())) {
804  std::stringstream newVarName;
805  newVarName << "_"
806  << "&isin;" << types.find(var)->second;
807  return std::make_unique<AstVariable>(newVarName.str());
808  }
809  node->apply(*this);
810  return node;
811  }
812  };
813 
814  /* Note:
815  * Because the type of each argument is stored in the form [address -> type-set],
816  * the type-analysis result does not immediately apply to the clone due to differing
817  * addresses.
818  * Two ways around this:
819  * (1) Perform the type-analysis again for the cloned clause
820  * (2) Keep track of the addresses of equivalent arguments in the cloned clause
821  * Method (2) was chosen to avoid having to recompute the analysis each time.
822  */
823  AstClause* annotatedClause = clause->clone();
824 
825  // Maps x -> y, where x is the address of an argument in the original clause, and y
826  // is the address of the equivalent argument in the clone.
827  std::map<const AstArgument*, const AstArgument*> memoryMap;
828 
829  std::vector<const AstArgument*> originalAddresses;
830  visitDepthFirst(*clause, [&](const AstArgument& arg) { originalAddresses.push_back(&arg); });
831 
832  std::vector<const AstArgument*> cloneAddresses;
833  visitDepthFirst(*annotatedClause, [&](const AstArgument& arg) { cloneAddresses.push_back(&arg); });
834 
835  assert(cloneAddresses.size() == originalAddresses.size());
836 
837  for (size_t i = 0; i < originalAddresses.size(); i++) {
838  memoryMap[originalAddresses[i]] = cloneAddresses[i];
839  }
840 
841  // Map the types to the clause clone
842  std::map<const AstArgument*, TypeSet> cloneArgumentTypes;
843  for (auto& pair : argumentTypes) {
844  cloneArgumentTypes[memoryMap[pair.first]] = pair.second;
845  }
846 
847  // Create the type-annotated clause
848  TypeAnnotator annotator(cloneArgumentTypes);
849  annotatedClause->apply(annotator);
850  return annotatedClause;
851 }

853 void TypeAnalysis::run(const AstTranslationUnit& translationUnit) {
854  auto* typeEnvAnalysis = translationUnit.getAnalysis<TypeEnvironmentAnalysis>();
855  for (const AstRelation* rel : translationUnit.getProgram()->getRelations()) {
856  for (const AstClause* clause : rel->getClauses()) {
857  // Perform the type analysis
858  std::map<const AstArgument*, TypeSet> clauseArgumentTypes = analyseTypes(
859  typeEnvAnalysis->getTypeEnvironment(), *clause, translationUnit.getProgram());
860  argumentTypes.insert(clauseArgumentTypes.begin(), clauseArgumentTypes.end());
861 
862  if (!Global::config().get("debug-report").empty()) {
863  // Store an annotated clause for printing purposes
864  AstClause* annotatedClause = createAnnotatedClause(clause, clauseArgumentTypes);
865  annotatedClauses.push_back(std::unique_ptr<AstClause>(annotatedClause));
866  }
867  }
868  }

877 std::map<const AstArgument*, TypeSet> TypeAnalysis::analyseTypes(
878  const TypeEnvironment& env, const AstClause& clause, const AstProgram* program, bool verbose) {
879  struct Analysis : public AstConstraintAnalysis<TypeVar> {
880  const TypeEnvironment& env;
881  const AstProgram* program;
882  std::set<const AstAtom*> negated;
883 
884  Analysis(const TypeEnvironment& env, const AstProgram* program) : env(env), program(program) {}
885 
886  // predicate
887  void visitAtom(const AstAtom& atom) override {
888  // get relation
889  auto rel = getAtomRelation(&atom, program);
890  if (!rel) {
891  return; // error in input program
892  }
893 
894  auto atts = rel->getAttributes();
895  auto args = atom.getArguments();
896  if (atts.size() != args.size()) {
897  return; // error in input program
898  }
899 
900  // set upper boundary of argument types
901  for (unsigned i = 0; i < atts.size(); i++) {
902  const auto& typeName = atts[i]->getTypeName();
903  if (env.isType(typeName)) {
904  // check whether atom is not negated
905  if (negated.find(&atom) == negated.end()) {
906  addConstraint(isSubtypeOf(getVar(args[i]), env.getType(typeName)));
907  } else {
908  addConstraint(isSupertypeOf(getVar(args[i]), env.getType(typeName)));
909  }
910  }
911  }
912  }
913 
914  // #2 - negations need to be skipped
915  void visitNegation(const AstNegation& cur) override {
916  // add nested atom to black-list
917  negated.insert(cur.getAtom());
918  }
919 
920  // symbol
921  void visitStringConstant(const AstStringConstant& cnst) override {
922  // this type has to be a sub-type of symbol
923  addConstraint(isSubtypeOf(getVar(cnst), env.getSymbolType()));
924  }
925 
926  // number
927  void visitNumberConstant(const AstNumberConstant& cnst) override {
928  // this type has to be a sub-type of number
929  addConstraint(isSubtypeOf(getVar(cnst), env.getNumberType()));
930  }
931 
932  // binary constraint
933  void visitBinaryConstraint(const AstBinaryConstraint& rel) override {
934  auto lhs = getVar(rel.getLHS());
935  auto rhs = getVar(rel.getRHS());
936  addConstraint(isSubtypeOf(lhs, rhs));
937  addConstraint(isSubtypeOf(rhs, lhs));
938  }
939 
940  // unary functor
941  void visitUnaryFunctor(const AstUnaryFunctor& fun) override {
942  auto out = getVar(fun);
943  auto in = getVar(fun.getOperand());
944 
945  // add a constraint for the return type of the unary functor
946  if (fun.isNumerical()) {
947  addConstraint(isSubtypeOf(out, env.getNumberType()));
948  }
949  if (fun.isSymbolic()) {
950  addConstraint(isSubtypeOf(out, env.getSymbolType()));
951  }
952 
953  // add a constraint for the argument type of the unary functor
954  if (fun.acceptsNumbers()) {
955  addConstraint(isSubtypeOf(in, env.getNumberType()));
956  }
957  if (fun.acceptsSymbols()) {
958  addConstraint(isSubtypeOf(in, env.getSymbolType()));
959  }
960  }
961 
962  // binary functor
963  void visitBinaryFunctor(const AstBinaryFunctor& fun) override {
964  auto cur = getVar(fun);
965  auto lhs = getVar(fun.getLHS());
966  auto rhs = getVar(fun.getRHS());
967 
968  // add a constraint for the return type of the binary functor
969  if (fun.isNumerical()) {
970  addConstraint(isSubtypeOf(cur, env.getNumberType()));
971  }
972  if (fun.isSymbolic()) {
973  addConstraint(isSubtypeOf(cur, env.getSymbolType()));
974  }
975 
976  // add a constraint for the first argument of the binary functor
977  if (fun.acceptsNumbers(0)) {
978  addConstraint(isSubtypeOf(lhs, env.getNumberType()));
979  }
980  if (fun.acceptsSymbols(0)) {
981  addConstraint(isSubtypeOf(lhs, env.getSymbolType()));
982  }
983 
984  // add a constraint for the second argument of the binary functor
985  if (fun.acceptsNumbers(1)) {
986  addConstraint(isSubtypeOf(rhs, env.getNumberType()));
987  }
988  if (fun.acceptsSymbols(1)) {
989  addConstraint(isSubtypeOf(rhs, env.getSymbolType()));
990  }
991  }
992 
993  // ternary functor
994  void visitTernaryFunctor(const AstTernaryFunctor& fun) override {
995  auto cur = getVar(fun);
996 
997  auto a0 = getVar(fun.getArg(0));
998  auto a1 = getVar(fun.getArg(1));
999  auto a2 = getVar(fun.getArg(2));
1000 
1001  // add a constraint for the return type of the ternary functor
1002  if (fun.isNumerical()) {
1003  addConstraint(isSubtypeOf(cur, env.getNumberType()));
1004  }
1005  if (fun.isSymbolic()) {
1006  addConstraint(isSubtypeOf(cur, env.getSymbolType()));
1007  }
1008 
1009  // add a constraint for the first argument of the ternary functor
1010  if (fun.acceptsNumbers(0)) {
1011  addConstraint(isSubtypeOf(a0, env.getNumberType()));
1012  }
1013  if (fun.acceptsSymbols(0)) {
1014  addConstraint(isSubtypeOf(a0, env.getSymbolType()));
1015  }
1016 
1017  // add a constraint for the second argument of the ternary functor
1018  if (fun.acceptsNumbers(1)) {
1019  addConstraint(isSubtypeOf(a1, env.getNumberType()));
1020  }
1021  if (fun.acceptsSymbols(1)) {
1022  addConstraint(isSubtypeOf(a1, env.getSymbolType()));
1023  }
1024 
1025  // add a constraint for the third argument of the ternary functor
1026  if (fun.acceptsNumbers(2)) {
1027  addConstraint(isSubtypeOf(a2, env.getNumberType()));
1028  }
1029  if (fun.acceptsSymbols(2)) {
1030  addConstraint(isSubtypeOf(a2, env.getSymbolType()));
1031  }
1032  }
1033 
1034  // counter
1035  void visitCounter(const AstCounter& counter) override {
1036  // this value must be a number value
1037  addConstraint(isSubtypeOf(getVar(counter), env.getNumberType()));
1038  }
1039 
1040  // components of records
1041  void visitRecordInit(const AstRecordInit& init) override {
1042  // link element types with sub-values
1043  auto rec = getVar(init);
1044  int i = 0;
1045 
1046  for (const AstArgument* value : init.getArguments()) {
1047  addConstraint(isSubtypeOfComponent(getVar(value), rec, i++));
1048  }
1049  }
1050 
1051  // visit aggregates
1052  void visitAggregator(const AstAggregator& agg) override {
1053  // this value must be a number value
1054  addConstraint(isSubtypeOf(getVar(agg), env.getNumberType()));
1055 
1056  // also, the target expression needs to be a number
1057  if (auto expr = agg.getTargetExpression()) {
1058  addConstraint(isSubtypeOf(getVar(expr), env.getNumberType()));
1059  }
1060  }
1061  };
1062 
1063  // run analysis
1064  return Analysis(env, program).analyse(clause, verbose);
1065 }
