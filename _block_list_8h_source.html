27 /**
28  * A concurrent list data structure, which is implemented through a chunked linked-list.
29  * On their own, get/push_back are threadsafe, however clearing/destructing is undefined
30  * behaviour if get/push_backs are in progress.
31  */
32 template <class T>
33 class concurrent_list {
34  // how large each cv_data->m_data array is
35  const size_t chunk_size = 1000;
36 
37  struct cv_data {
38  // how many elements are currently in m_data
39  std::atomic<size_t> m_size;
40  T* m_data;
41  // ptr to the next cv_data
42  std::atomic<cv_data*> next;
43  };

45  // number of elements in total

63  concurrent_list(concurrent_list&& other) : concurrent_list() {

114  /*
115  * Append the value onto the end of the list, allocating more space if required.
116  * @param val the value to insert
117  * thread safe for many writers
118  */
119  void push_back(T val) {
120  std::unique_lock<std::mutex> lk(writeMutex);
121 
122  cv_data* curr = cData.load();
123  // fast forward curr until non-null
124  while (curr->next.load() != nullptr) curr = curr->next.load();
125 
126  cv_data* shadow = nullptr;
127  cv_data* front = curr;
128 
129  // do we need to expand?
130  if (curr->m_size == chunk_size) {
131  shadow = new cv_data;
132  shadow->m_size = 0;
133  shadow->m_data = new T[chunk_size];
134  shadow->next = nullptr;
135 
136  front = shadow;
137  }
138 
139  front->m_data[front->m_size] = val;
140  front->m_size += 1;
141  if (shadow != nullptr) curr->next = shadow;
142  container_size += 1;
143  }

174  cv_data* curr = cData;

184  curr->m_size = 0;
185  curr->m_data = new T[chunk_size];
186  curr->next = nullptr;
187 
188  container_size = 0;
189 
190  cData = curr;
191  }

200 /**
201  * A class that is designed to mimic std::list, but with better destructor speed
202  * It achieves this by allocating comparatively large chunks of memory as it needs
203  * This is not thread safe, except, when there is guarantee of at most one writer AND
204  * the TBB data structure is in use.
205  *
206  */
207 template <class T>
208 class BlockList {

210 
211  mutable size_t m_size = 0;
212 
213 public:
214  BlockList() : listData() {}

216  /** copy constructor */
217  BlockList(const BlockList& other) {
218  size_t othblocks = other.listData.size();
219  for (size_t i = 0; i < othblocks; ++i) {
220  listData.push_back(new T[BLOCKSIZE]);

222  }
223  this->m_size = other.m_size;
224  }

253  /**
254  * Add a value to the blocklist
255  * @param val : value to be added
256  */
257  void add(const T& val) {
258  // store it in the vector
259  size_t blocknum = m_size >> BLOCKBITS;
260  size_t blockindex = m_size & (BLOCKSIZE - 1);
261 
262  // we need to add a new block
263  if (blockindex == 0) {
264  this->listData.push_back(new T[BLOCKSIZE]);
265  }
266 
267  this->listData[blocknum][blockindex] = T(val);
268 
269  ++m_size;
270  }

297  /**
298  * Remove the last element from the data
299  * and destroying the block if its the last one in it
300  * @return the copied value
301  */
302  T pop() {
303  if (m_size == 0) throw std::runtime_error("pop called on empty blocklist");
304 
305  --m_size;
306  size_t blocknum = m_size >> BLOCKBITS;
307  size_t blockindex = m_size & (BLOCKSIZE - 1);
308 
309  // we need to destroy the block - we're the last one standing
310  if (blockindex == 0) {
311  T retVal = listData[blocknum][blockindex];
312 
313  delete[] listData[blocknum];
314 
315  listData.pop_back();
316 
317  return retVal;
318 
319  } else {
320  return listData[blocknum][blockindex];
321  }
322  }

324  /**
325  * Move the other list's elements here
326  * REMOVING them from the other
327  * Similar to std::list.splice()
328  * @param other the other list
329  */
330  void niptuck(BlockList& other) {
331  if (other.size() == 0) return;
332 
333  // fill up the remainder of this BLs last block with elements from the end of other's
334  size_t requiredEmpties = BLOCKSIZE - (m_size & (BLOCKSIZE - 1));
335  // we ignore the case of having to remove the entire block - its more efficient to do later
336  if (requiredEmpties != BLOCKSIZE) {
337  // pop and add
338  for (size_t rem = requiredEmpties; rem > 0 && other.size() > 0; --rem) {
339  this->add(other.pop());
340  }
341  }
342 
343  // negative values mess up the shift
344  if (other.size() == 0) {
345  other.listData.clear();
346  other.m_size = 0;
347  return;
348  }
349 
350  // if we have remainder, we've got to make sure that we simply plonk in the pointers
351  // size-1, because BLOCKSIZE >> BLOCKBITS should equal 1
352  size_t otherblocks = (other.m_size - 1) >> BLOCKBITS;
353  ++otherblocks;
354  for (size_t i = 0; i < otherblocks; ++i) {
355  listData.push_back(other.listData[i]);
356  }
357 
358  this->m_size += other.m_size;
359  other.listData.clear();
360  other.m_size = 0;
361  }
