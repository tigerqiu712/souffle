75 void RedundantRelations::run(const AstTranslationUnit& translationUnit) {
76  precedenceGraph = translationUnit.getAnalysis<PrecedenceGraph>();
77 
78  std::set<const AstRelation*> work;
79  std::set<const AstRelation*> notRedundant;
80 
81  const std::vector<AstRelation*>& relations = translationUnit.getProgram()->getRelations();
82 
83  /* Add all output relations to the work set */
84  for (const AstRelation* r : relations) {
85  if (r->isComputed()) {
86  work.insert(r);
87  }
88  }
89 
90  /* Find all relations which are not redundant for the computations of the
91  output relations. */
92  while (!work.empty()) {
93  /* Chose one element in the work set and add it to notRedundant */
94  const AstRelation* u = *(work.begin());
95  work.erase(work.begin());
96  notRedundant.insert(u);
97 
98  /* Find all predecessors of u and add them to the worklist
99  if they are not in the set notRedundant */
100  for (const AstRelation* predecessor : precedenceGraph->graph().predecessors(u)) {
101  if (!notRedundant.count(predecessor)) {
102  work.insert(predecessor);
103  }
104  }
105  }
106 
107  /* All remaining relations are redundant. */
108  redundantRelations.clear();
109  for (const AstRelation* r : relations) {
110  if (!notRedundant.count(r)) {
111  redundantRelations.insert(r);
112  }
113  }
114 }

132 bool RecursiveClauses::computeIsRecursive(
133  const AstClause& clause, const AstTranslationUnit& translationUnit) const {
134  const AstProgram& program = *translationUnit.getProgram();
135 
136  // we want to reach the atom of the head through the body
137  const AstRelation* trg = getHeadRelation(&clause, &program);
138 
139  std::set<const AstRelation*> reached;
140  std::vector<const AstRelation*> worklist;
141 
142  // set up start list
143  for (const AstAtom* cur : clause.getAtoms()) {
144  auto rel = program.getRelation(cur->getName());
145  if (rel == trg) {
146  return true;
147  }
148  worklist.push_back(rel);
149  }
150 
151  // process remaining elements
152  while (!worklist.empty()) {
153  // get next to process
154  const AstRelation* cur = worklist.back();
155  worklist.pop_back();
156 
157  // skip null pointers (errors in the input code)
158  if (!cur) {
159  continue;
160  }
161 
162  // check whether this one has been checked before
163  if (!reached.insert(cur).second) {
164  continue;
165  }
166 
167  // check all atoms in the relations
168  for (const AstClause* cl : cur->getClauses()) {
169  for (const AstAtom* at : cl->getAtoms()) {
170  auto rel = program.getRelation(at->getName());
171  if (rel == trg) {
172  return true;
173  }
174  worklist.push_back(rel);
175  }
176  }
177  }
178 
179  // no cycles found
180  return false;
181 }

183 void SCCGraph::run(const AstTranslationUnit& translationUnit) {
184  precedenceGraph = translationUnit.getAnalysis<PrecedenceGraph>();
185  sccToRelation.clear();
186  relationToScc.clear();
187  predecessors.clear();
188  successors.clear();
189 
190  /* Compute SCC */
191  std::vector<AstRelation*> relations = translationUnit.getProgram()->getRelations();
192  size_t counter = 0;
193  size_t numSCCs = 0;
194  std::stack<const AstRelation*> S, P;
195  std::map<const AstRelation*, size_t> preOrder; // Pre-order number of a node (for Gabow's Algo)
196  for (const AstRelation* relation : relations) {
197  relationToScc[relation] = preOrder[relation] = (size_t)-1;
198  }
199  for (const AstRelation* relation : relations) {
200  if (preOrder[relation] == (size_t)-1) {
201  scR(relation, preOrder, counter, S, P, numSCCs);
202  }
203  }
204 
205  /* Build SCC graph */
206  successors.resize(numSCCs);
207  predecessors.resize(numSCCs);
208  for (const AstRelation* u : relations) {
209  for (const AstRelation* v : precedenceGraph->graph().predecessors(u)) {
210  auto scc_u = relationToScc[u];
211  auto scc_v = relationToScc[v];
212  assert(scc_u < numSCCs && "Wrong range");
213  assert(scc_v < numSCCs && "Wrong range");
214  if (scc_u != scc_v) {
215  predecessors[scc_u].insert(scc_v);
216  successors[scc_v].insert(scc_u);
217  }
218  }
219  }
220 
221  /* Store the relations for each SCC */
222  sccToRelation.resize(numSCCs);
223  for (const AstRelation* relation : relations) {
224  sccToRelation[relationToScc[relation]].insert(relation);
225  }
226 }

228 /* Compute strongly connected components using Gabow's algorithm (cf. Algorithms in
229  * Java by Robert Sedgewick / Part 5 / Graph * algorithms). The algorithm has linear
230  * runtime. */
231 void SCCGraph::scR(const AstRelation* w, std::map<const AstRelation*, size_t>& preOrder, size_t& counter,
232  std::stack<const AstRelation*>& S, std::stack<const AstRelation*>& P, size_t& numSCCs) {
233  preOrder[w] = counter++;
234  S.push(w);
235  P.push(w);
236  for (const AstRelation* t : precedenceGraph->graph().predecessors(w)) {
237  if (preOrder[t] == (size_t)-1) {
238  scR(t, preOrder, counter, S, P, numSCCs);
239  } else if (relationToScc[t] == (size_t)-1) {
240  while (preOrder[P.top()] > preOrder[t]) {
241  P.pop();
242  }
243  }
244  }
245  if (P.top() == w) {
246  P.pop();
247  } else {
248  return;
249  }
250 
251  const AstRelation* v;
252  do {
253  v = S.top();
254  S.pop();
255  relationToScc[v] = numSCCs;
256  } while (v != w);
257  numSCCs++;
258 }

279 int TopologicallySortedSCCGraph::topologicalOrderingCost(const std::vector<size_t>& permutationOfSCCs) const {
280  // create variables to hold the cost of the current SCC and the permutation as a whole
281  int costOfSCC = 0;
282  int costOfPermutation = -1;
283  // obtain an iterator to the end of the already ordered partition of sccs
284  auto it_k = permutationOfSCCs.begin() + sccOrder.size();
285  // for each of the scc's in the ordering, resetting the cost of the scc to zero on each loop
286  for (auto it_i = permutationOfSCCs.begin(); it_i != permutationOfSCCs.end(); ++it_i, costOfSCC = 0) {
287  // if the index of the current scc is after the end of the ordered partition
288  if (it_i >= it_k) {
289  // check that the index of all predecessor sccs of are before the index of the current scc
290  for (auto scc : sccGraph->getPredecessorSCCs(*it_i)) {
291  if (std::find(permutationOfSCCs.begin(), it_i, scc) == it_i) {
292  // if not, the sort is not a valid topological sort
293  return -1;
294  }
295  }
296  }
297  // otherwise, calculate the cost of the current scc
298  // as the number of sccs with an index before the current scc
299  for (auto it_j = permutationOfSCCs.begin(); it_j != it_i; ++it_j) {
300  // having some successor scc with an index after the current scc
301  for (auto scc : sccGraph->getSuccessorSCCs(*it_j)) {
302  if (std::find(permutationOfSCCs.begin(), it_i, scc) == it_i) {
303  costOfSCC++;
304  }
305  }
306  }
307  // and if this cost is greater than the maximum recorded cost for the whole permutation so far,
308  // set the cost of the permutation to it
309  if (costOfSCC > costOfPermutation) {
310  costOfPermutation = costOfSCC;
311  }
312  }
313  return costOfPermutation;
314 }

316 void TopologicallySortedSCCGraph::computeTopologicalOrdering(size_t scc, std::vector<bool>& visited) {
317  // create a flag to indicate that a successor was visited (by default it hasn't been)
318  bool found = false, hasUnvisitedSuccessor = false, hasUnvisitedPredecessor = false;
319  // for each successor of the input scc
320  const auto& successorsToVisit = sccGraph->getSuccessorSCCs(scc);
321  for (const auto scc_i : successorsToVisit) {
322  if (visited[scc_i]) {
323  continue;
324  }
325  hasUnvisitedPredecessor = false;
326  const auto& successorsPredecessors = sccGraph->getPredecessorSCCs(scc_i);
327  for (const auto scc_j : successorsPredecessors) {
328  if (!visited[scc_j]) {
329  hasUnvisitedPredecessor = true;
330  break;
331  }
332  }
333  if (!hasUnvisitedPredecessor) {
334  // give it a temporary marking
335  visited[scc_i] = true;
336  // add it to the permanent ordering
337  sccOrder.push_back(scc_i);
338  // and use it as a root node in a recursive call to this function
339  computeTopologicalOrdering(scc_i, visited);
340  // finally, indicate that a successor has been found for this node
341  found = true;
342  }
343  }
344  // return at once if no valid successors have been found; as either it has none or they all have a
345  // better predecessor
346  if (!found) {
347  return;
348  }
349  hasUnvisitedPredecessor = false;
350  const auto& predecessors = sccGraph->getPredecessorSCCs(scc);
351  for (const auto scc_j : predecessors) {
352  if (!visited[scc_j]) {
353  hasUnvisitedPredecessor = true;
354  break;
355  }
356  }
357  hasUnvisitedSuccessor = false;
358  const auto& successors = sccGraph->getSuccessorSCCs(scc);
359  for (const auto scc_j : successors) {
360  if (!visited[scc_j]) {
361  hasUnvisitedSuccessor = true;
362  break;
363  }
364  }
365  // otherwise, if more white successors remain for the current scc, use it again as the root node in a
366  // recursive call to this function
367  if (hasUnvisitedSuccessor && !hasUnvisitedPredecessor) {
368  computeTopologicalOrdering(scc, visited);
369  }
370 }

372 void TopologicallySortedSCCGraph::run(const AstTranslationUnit& translationUnit) {
373  // obtain the scc graph
374  sccGraph = translationUnit.getAnalysis<SCCGraph>();
375  // clear the list of ordered sccs
376  sccOrder.clear();
377  std::vector<bool> visited;
378  visited.resize(sccGraph->getNumberOfSCCs());
379  std::fill(visited.begin(), visited.end(), false);
380  // generate topological ordering using forwards algorithm (like Khan's algorithm)
381  // for each of the sccs in the graph
382  for (size_t scc = 0; scc < sccGraph->getNumberOfSCCs(); ++scc) {
383  // if that scc has no predecessors
384  if (sccGraph->getPredecessorSCCs(scc).empty()) {
385  // put it in the ordering
386  sccOrder.push_back(scc);
387  visited[scc] = true;
388  // if the scc has successors
389  if (!sccGraph->getSuccessorSCCs(scc).empty()) {
390  computeTopologicalOrdering(scc, visited);
391  }
392  }
393  }
394 }

458 std::vector<std::set<const AstRelation*>> RelationSchedule::computeRelationExpirySchedule(
459  const AstTranslationUnit& translationUnit) {
460  std::vector<std::set<const AstRelation*>> relationExpirySchedule;
461  /* Compute for each step in the reverse topological order
462  of evaluating the SCC the set of alive relations. */
463  size_t numSCCs = topsortSCCGraph->order().size();
464 
465  /* Alive set for each step */
466  std::vector<std::set<const AstRelation*>> alive(numSCCs);
467  /* Resize expired relations sets */
468  relationExpirySchedule.resize(numSCCs);
469  const auto& sccGraph = translationUnit.getAnalysis<SCCGraph>();
470 
471  /* Compute all alive relations by iterating over all steps in reverse order
472  determine the dependencies */
473  for (size_t orderedSCC = 1; orderedSCC < numSCCs; orderedSCC++) {
474  /* Add alive set of previous step */
475  alive[orderedSCC].insert(alive[orderedSCC - 1].begin(), alive[orderedSCC - 1].end());
476 
477  /* Add predecessors of relations computed in this step */
478  auto scc = topsortSCCGraph->order()[numSCCs - orderedSCC];
479  for (const AstRelation* r : sccGraph->getInternalRelations(scc)) {
480  for (const AstRelation* predecessor : precedenceGraph->graph().predecessors(r)) {
481  alive[orderedSCC].insert(predecessor);
482  }
483  }
484 
485  /* Compute expired relations in reverse topological order using the set difference of the alive sets
486  between steps. */
487  std::set_difference(alive[orderedSCC].begin(), alive[orderedSCC].end(), alive[orderedSCC - 1].begin(),
488  alive[orderedSCC - 1].end(),
489  std::inserter(relationExpirySchedule[numSCCs - orderedSCC],
490  relationExpirySchedule[numSCCs - orderedSCC].end()));
491  }
492 
493  return relationExpirySchedule;
494 }
