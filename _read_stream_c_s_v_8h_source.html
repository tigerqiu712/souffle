54  * Read and return the next tuple.
55  *
56  * Returns nullptr if no tuple was readable.
57  * @return
58  */
59  std::unique_ptr<RamDomain[]> readNextTuple() override {
60  if (file.eof()) {
61  return nullptr;
62  }
63  std::string line;
64  std::unique_ptr<RamDomain[]> tuple = std::make_unique<RamDomain[]>(symbolMask.getArity());
65  bool error = false;
66 
67  if (!getline(file, line)) {
68  return nullptr;
69  }
70  // Handle Windows line endings on non-Windows systems
71  if (line.back() == '\r') {
72  line = line.substr(0, line.length() - 1);
73  }
74  ++lineNumber;
75 
76  size_t start = 0, end = 0, columnsFilled = 0;
77  for (uint32_t column = 0; end < line.length(); column++) {
78  if (isProvenance && columnsFilled >= symbolMask.getArity() - 2) {
79  end = line.length();
80  break;
81  }
82 
83  end = line.find(delimiter, start);
84  if (end == std::string::npos) {
85  end = line.length();
86  }
87  std::string element;
88  if (start <= end && end <= line.length()) {
89  element = line.substr(start, end - start);
90  if (element.empty()) {
91  element = "n/a";
92  }
93  } else {
94  if (!error) {
95  std::stringstream errorMessage;
96  errorMessage << "Value missing in column " << column + 1 << " in line " << lineNumber
97  << "; ";
98  throw std::invalid_argument(errorMessage.str());
99  }
100  element = "n/a";
101  }
102  start = end + delimiter.size();
103  if (inputMap.count(column) == 0) {
104  continue;
105  }
106  ++columnsFilled;
107  if (symbolMask.isSymbol(column)) {
108  tuple[inputMap[column]] = symbolTable.unsafeLookup(element);
109  } else {
110  try {
111 #if RAM_DOMAIN_SIZE == 64
112  tuple[inputMap[column]] = std::stoll(element);
113 #else
114  tuple[inputMap[column]] = std::stoi(element);
115 #endif
116  } catch (...) {
117  if (!error) {
118  std::stringstream errorMessage;
119  errorMessage << "Error converting number <" + element + "> in column " << column + 1
120  << " in line " << lineNumber << "; ";
121  throw std::invalid_argument(errorMessage.str());
122  }
123  }
124  }
125  }
126 
127  // add two provenance columns
128  if (isProvenance) {
129  tuple[symbolMask.getArity() - 2] = 0;
130  tuple[symbolMask.getArity() - 1] = 0;
131 
132  if (columnsFilled >= symbolMask.getArity() - 2 && columnsFilled <= symbolMask.getArity()) {
133  columnsFilled = symbolMask.getArity();
134  }
135  // columnsFilled += 2;
136  }
137 
138  if (columnsFilled != symbolMask.getArity()) {
139  std::stringstream errorMessage;
140  errorMessage << "Values missing in line " << lineNumber << "; ";
141  throw std::invalid_argument(errorMessage.str());
142  }
143  if (end != line.length()) {
144  if (!error) {
145  std::stringstream errorMessage;
146  errorMessage << "Too many cells in line " << lineNumber << "; ";
147  throw std::invalid_argument(errorMessage.str());
148  }
149  }
150  if (error) {
151  throw std::invalid_argument("cannot parse fact file");
152  }
153 
154  return tuple;
155  }
156 
157  std::string getDelimiter(const IODirectives& ioDirectives) const {
158  if (ioDirectives.has("delimiter")) {
159  return ioDirectives.get("delimiter");
160  }
161  return "\t";
162  }

164  std::map<int, int> getInputColumnMap(const IODirectives& ioDirectives, const unsigned arity) const {
165  std::string columnString = "";
166  if (ioDirectives.has("columns")) {
167  columnString = ioDirectives.get("columns");
168  }
169  std::map<int, int> inputMap;
170 
171  if (!columnString.empty()) {
172  std::istringstream iss(columnString);
173  std::string mapping;
174  int index = 0;
175  while (std::getline(iss, mapping, ':')) {
176  inputMap[stoi(mapping)] = index++;
177  }
178  if (inputMap.size() < arity) {
179  throw std::invalid_argument("Invalid column set was given: <" + columnString + ">");
180  }
181  } else {
182  while (inputMap.size() < arity) {
183  int size = inputMap.size();
184  inputMap[size] = size;
185  }
186  }
187  return inputMap;
188  }
