99 Table inline OutputProcessor::getRelTable() {
100  std::unordered_map<std::string, std::shared_ptr<Relation>>& relation_map = programRun->getRelation_map();
101  Table table;
102  for (auto& rel : relation_map) {
103  std::shared_ptr<Relation> r = rel.second;
104  Row row(12);
105  double total_time = r->getNonRecTime() + r->getRecTime() + r->getCopyTime();
106  row[0] = std::shared_ptr<CellInterface>(new Cell<double>(total_time));
107  row[1] = std::shared_ptr<CellInterface>(new Cell<double>(r->getNonRecTime()));
108  row[2] = std::shared_ptr<CellInterface>(new Cell<double>(r->getRecTime()));
109  row[3] = std::shared_ptr<CellInterface>(new Cell<double>(r->getCopyTime()));
110  row[4] = std::shared_ptr<CellInterface>(new Cell<long>(r->getNum_tuplesRel()));
111  row[5] = std::shared_ptr<CellInterface>(new Cell<std::string>(r->getName()));
112  row[6] = std::shared_ptr<CellInterface>(new Cell<std::string>(r->getId()));
113  row[7] = std::shared_ptr<CellInterface>(new Cell<std::string>(r->getLocator()));
114  if (total_time != 0.0) {
115  row[8] = std::shared_ptr<CellInterface>(
116  new Cell<long>(r->getNum_tuplesRel() / (total_time * 1000)));
117  } else {
118  row[8] = std::shared_ptr<CellInterface>(new Cell<long>(r->getNum_tuplesRel() / 1.0));
119  }
120  row[9] = std::shared_ptr<CellInterface>(new Cell<double>(r->getLoadtime()));
121  row[10] = std::shared_ptr<CellInterface>(new Cell<double>(r->getSavetime()));
122  row[11] = std::shared_ptr<CellInterface>(new Cell<long>(r->getMaxRSSDiff()));
123 
124  table.addRow(std::make_shared<Row>(row));
125  }
126  return table;
127 }

157  row[7] = std::shared_ptr<CellInterface>(new Cell<std::string>(rel.second->getName()));
158  row[8] = std::shared_ptr<CellInterface>(new Cell<long>(0));

297 Table inline OutputProcessor::getVersions(std::string strRel, std::string strRul) {
298  std::unordered_map<std::string, std::shared_ptr<Relation>>& relation_map = programRun->getRelation_map();
299  std::unordered_map<std::string, std::shared_ptr<Row>> rule_map;
300 
301  double tot_rec_tup = programRun->getTotNumRecTuples();
302  double tot_copy_time = programRun->getTotCopyTime();
303 
304  for (auto& _rel : relation_map) {
305  std::shared_ptr<Relation> rel = _rel.second;
306  if (rel->getId().compare(strRel) == 0) {
307  for (auto& iter : rel->getIterations()) {
308  for (auto& _rul : iter->getRul_rec()) {
309  std::shared_ptr<Rule> rul = _rul.second;
310  if (rul->getId().compare(strRul) == 0) {
311  std::string strTemp =
312  rul->getName() + rul->getLocator() + std::to_string(rul->getVersion());
313 
314  if (rule_map.find(strTemp) != rule_map.end()) {
315  std::shared_ptr<Row> _row = rule_map[strTemp];
316  Row row = *_row;
317  row[2] = std::shared_ptr<CellInterface>(
318  new Cell<double>(row[2]->getDoubVal() + rul->getRuntime()));
319  row[4] = std::shared_ptr<CellInterface>(
320  new Cell<long>(row[4]->getLongVal() + rul->getNum_tuples()));
321  row[0] = std::shared_ptr<CellInterface>(new Cell<double>(rul->getRuntime()));
322  rule_map[strTemp] = std::make_shared<Row>(row);
323  } else {
324  Row row(10);
325  row[1] = std::shared_ptr<CellInterface>(new Cell<double>(0.0));
326  row[2] = std::shared_ptr<CellInterface>(new Cell<double>(rul->getRuntime()));
327  row[4] = std::shared_ptr<CellInterface>(new Cell<long>(rul->getNum_tuples()));
328  row[5] = std::shared_ptr<CellInterface>(new Cell<std::string>(rul->getName()));
329  row[6] = std::shared_ptr<CellInterface>(new Cell<std::string>(rul->getId()));
330  row[7] = std::shared_ptr<CellInterface>(new Cell<std::string>(rel->getName()));
331  row[8] = std::shared_ptr<CellInterface>(new Cell<long>(rul->getVersion()));
332  row[9] = std::shared_ptr<CellInterface>(new Cell<std::string>(rul->getLocator()));
333  row[0] = std::shared_ptr<CellInterface>(new Cell<double>(rul->getRuntime()));
334  rule_map[strTemp] = std::make_shared<Row>(row);
335  }
336  }
337  }
338  }
339  for (auto row : rule_map) {
340  Row t = *row.second;
341  double d = tot_rec_tup;
342  long d2 = t[4]->getLongVal();
343  t[3] = std::shared_ptr<CellInterface>(new Cell<double>(d2 * tot_copy_time / d));
344  t[0] = std::shared_ptr<CellInterface>(
345  new Cell<double>(t[1]->getDoubVal() + t[2]->getDoubVal() + t[3]->getDoubVal()));
346  rule_map[row.first] = std::make_shared<Row>(t);
347  }
348  break;
349  }
350  }
351 
352  Table table;
353  for (auto& _row : rule_map) {
354  table.addRow(_row.second);
355  }
356  return table;
357 }
