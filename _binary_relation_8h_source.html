395  /**
396  * Helper function to make the frontIter jump to the next valid Trie & update the current one
397  * @return whether we've reached end() or not
398  */
399  bool advanceFrontIter() {
400  // if we're at the end of this current Trie
401  if (frontIter == cTrie->end() || ++souffle::Trie<1>::iterator(frontIter) == cTrie->end()) {
402  // reaching the end of this trie means that the closure has completed
403  if (ityp == CLOSURE || ityp == FRONTPROD) {
404  isEndVal = true;
405  return true;
406  }
407 
408  // remove this trie from iterList as it means we've exhausted it
409  iterList.remove_if(
410  [&](std::pair<std::shared_ptr<souffle::Trie<1>>, souffle::Trie<1>::iterator> x) {
411  return x.first == cTrie;
412  });
413 
414  // have we exhausted all other tries?
415  if (iterList.empty()) {
416  isEndVal = true;
417  return true;
418  }
419 
420  // find the next smallest Trie
421  bool isStart = true;
422  std::pair<std::shared_ptr<souffle::Trie<1>>, souffle::Trie<1>::iterator>* ref = nullptr;
423 
424  for (auto& x : iterList) {
425  if (isStart || (*x.second)[0] < (*frontIter)[0]) {
426  ref = &x;
427  cTrie = x.first;
428  frontIter = souffle::Trie<1>::iterator(x.second);
429  isStart = false;
430  }
431  }
432  // as we've made frontIter point to this, we move our iter to the next one in the Trie
433  ++ref->second;
434  backIter = cTrie->begin();
435 
436  } else {
437  if (ityp == FRONTPROD) {
438  // no more front closure?
439  if (fronts.empty()) {
440  isEndVal = true;
441  return true;
442  }
443 
444  // step frontIter until we hit the next valid fronts
445  auto newFront = fronts.front();
446  fronts.pop_front();
447  do {
448  ++frontIter;
449  } while ((*frontIter)[0] != newFront);
450 
451  } else {
452  // we can just step frontIter along one, because it will not step past the end of a trie
453  ++frontIter;
454  // step the front reference pointer one along
455  for (auto& x : iterList) {
456  if (x.first == cTrie) {
457  ++x.second;
458  break;
459  }
460  }
461  }
462 
463  backIter = cTrie->begin();
464  }
465 
466  // we live to yield another time
467  return false;
468  }

704  * If that wasn't clear, so let's say there's an element e in start
705  * The iterator will generate the pairs (e, a), (e, b), ... (e, n) - assuming the elements a,b...n
706  * are in e's disjoint set
707  * Requires that all elements in start are within the same disjoint set (and in order)
708  * Changing this requires extensive changes to the underlying iterator(BinaryRelation*,
709  * std::vector<>) ctor
710  * @param start a list of front elements that must exist at the front pairing generated
711  * @return the resulting iterator that satisfies this
712  */
713  iterator frontProduct(std::list<DomainInt> start) const {
714  if (start.size() == 0) throw "invalid sized vector for front product";
715 
716  auto rep = sds.readOnlyFindNode(start.front());
717  if (std::any_of(start.begin(), start.end(),
718  [&](DomainInt i) { return sds.readOnlyFindNode(i) != rep; })) {
719  throw "elements not within same disjoint set";
720  }
721 
722  if (!std::is_sorted(start.begin(), start.end())) {
723  for (auto x : start) {
724  std::cout << x << '\t';
725  }
726  std::cout << std::endl;
727 
728  throw "elements are not sorted";
729  }
730  // the trie that contains every value in start
731  auto trie = generateTrieIfNone(rep);
732 
733  return iterator(this, start, trie);
734  }
735 
736  /**
737  * Begin an iterator over all pairs within a single disjoint set
738  * @param rep the representative of (or element within) a disjoint set of which to generate all pairs
739  * @return an iterator that will generate all pairs within the disjoint set
740  */
741  iterator closure(DomainInt rep) const {
742  // the trie that contains this rep
743  auto trie = generateTrieIfNone(rep);
744 
745  return iterator(this, rep, trie);
746  }
747 
748  /**
749  * Generate an approximate number of iterators for parallel iteration
750  * The iterators returned are not necessarily equal in size, but in practise are approximately similarly
751  * sized
752  * Depending on the structure of the data, there can be more or less partitions returned than requested.
753  * @param chunks the number of requested partitions
754  * @return a list of the iterators as ranges
755  */
756  std::vector<souffle::range<iterator>> partition(size_t chunks) const {
757  std::vector<souffle::range<iterator>> ret;
758 
759  for (auto rep = sds.beginReps(); rep != sds.endReps(); ++rep) {
760  generateTrieIfNone(*rep);
761  }
762  // num pairs
763  const size_t sz = this->size();
764 
765  // 0 or 1 chunks
766  if (chunks <= 1 || sz == 0) return {souffle::make_range(begin(), end())};
767 
768  // how many pairs can we fit within each iterator? (integer ceil division)
769  const size_t chunkSize = (sz + (chunks - 1)) / chunks;
770 
771  for (auto djSet = sds.beginReps(); djSet != sds.endReps(); ++djSet) {
772  const DomainInt rep = *djSet;
773 
774  const size_t djSetSize = sds.sizeOfRepresentativeSet(rep);
775  size_t cSize = 0;
776 
777  // will the entire djSet's pairs fit within a single iterator?
778  if (djSetSize * djSetSize <= chunkSize) {
779  // fit all pairs within this disjoint set into a single iterator
780  ret.push_back(souffle::make_range(closure(rep), end()));
781 
782  } else {
783  // add as many of this djSet closures into each iterator
784  std::list<DomainInt> fronts;
785 
786  for (auto el = sds.begin(rep); el != sds.end(rep); ++el) {
787  fronts.push_back(*el);
788  cSize += djSetSize;
789 
790  // iterator is full now? push this iterator set onto the return val
791  if (cSize >= chunkSize) {
792  fronts.sort();
793 
794  ret.push_back(souffle::make_range(frontProduct(fronts), end()));
795 
796  fronts.clear();
797 
798  cSize = 0;
799  }
800  }
801  // if there's any remainder still
802  if (cSize != 0) ret.push_back(souffle::make_range(frontProduct(fronts), end()));
803  }
804  }
805  return ret;
806  }
807 };
