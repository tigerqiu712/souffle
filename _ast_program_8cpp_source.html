145 /* Print program in textual format */
146 void AstProgram::print(std::ostream& os) const {
147  /* Print types */
148  os << "// ----- Types -----\n";
149  for (const auto& cur : types) {
150  os << *cur.second << "\n";
151  }
152 
153  /* Print components */
154  if (!components.empty()) {
155  os << "\n// ----- Components -----\n";
156  for (const auto& cur : components) {
157  os << *cur << "\n";
158  }
159  }
160 
161  /* Print instantiations */
162  if (!instantiations.empty()) {
163  os << "\n";
164  for (const auto& cur : instantiations) {
165  os << *cur << "\n";
166  }
167  }
168 
169  /* Print relations */
170  os << "\n// ----- Relations -----\n";
171  for (const auto& cur : relations) {
172  const std::unique_ptr<AstRelation>& rel = cur.second;
173  os << "\n\n// -- " << rel->getName() << " --\n";
174  os << *rel << "\n\n";
175  for (const auto clause : rel->getClauses()) {
176  os << *clause << "\n\n";
177  }
178  for (const auto ioDirective : rel->getIODirectives()) {
179  os << *ioDirective << "\n\n";
180  }
181  }
182 
183  if (!clauses.empty()) {
184  os << "\n// ----- Orphan Clauses -----\n";
185  os << join(clauses, "\n\n", print_deref<std::unique_ptr<AstClause>>()) << "\n";
186  }
187 
188  if (!ioDirectives.empty()) {
189  os << "\n// ----- Orphan IO directives -----\n";
190  os << join(ioDirectives, "\n\n", print_deref<std::unique_ptr<AstIODirective>>()) << "\n";
191  }
192 
193  if (!pragmaDirectives.empty()) {
194  os << "\n// ----- Pragma -----\n";
195  for (const auto& cur : pragmaDirectives) {
196  os << *cur << "\n";
197  }
198  }
199 }

201 AstProgram* AstProgram::clone() const {
202  // create copy
203  auto res = new AstProgram();
204 
205  // move types
206  for (const auto& cur : types) {
207  res->types.insert(std::make_pair(cur.first, std::unique_ptr<AstType>(cur.second->clone())));
208  }
209 
210  // move relations
211  for (const auto& cur : relations) {
212  res->relations.insert(std::make_pair(cur.first, std::unique_ptr<AstRelation>(cur.second->clone())));
213  }
214 
215  // move components
216  for (const auto& cur : components) {
217  res->components.push_back(std::unique_ptr<AstComponent>(cur->clone()));
218  }
219 
220  // move component instantiations
221  for (const auto& cur : instantiations) {
222  res->instantiations.push_back(std::unique_ptr<AstComponentInit>(cur->clone()));
223  }
224 
225  // move ioDirectives
226  for (const auto& cur : ioDirectives) {
227  res->ioDirectives.push_back(std::unique_ptr<AstIODirective>(cur->clone()));
228  }
229 
230  ErrorReport errors;
231 
232  res->finishParsing();
233 
234  // done
235  return res;

260 void AstProgram::finishParsing() {
261  // unbound clauses with no relation defined
262  std::vector<std::unique_ptr<AstClause>> unbound;
263 
264  // add clauses
265  for (auto& cur : clauses) {
266  auto pos = relations.find(cur->getHead()->getName());
267  if (pos != relations.end()) {
268  pos->second->addClause(std::move(cur));
269  } else {
270  unbound.push_back(std::move(cur));
271  }
272  }
273  // remember the remaining orphan clauses
274  clauses.clear();
275  clauses.swap(unbound);
276 
277  // unbound directives with no relation defined
278  std::vector<std::unique_ptr<AstIODirective>> unboundDirectives;
279 
280  // add IO directives
281  for (auto& cur : ioDirectives) {
282  auto pos = relations.find(cur->getName());
283  if (pos != relations.end()) {
284  pos->second->addIODirectives(std::move(cur));
285  } else {
286  unboundDirectives.push_back(std::move(cur));
287  }
288  }
289  // remember the remaining orphan directives
290  ioDirectives.clear();
291  ioDirectives.swap(unboundDirectives);
292 }
