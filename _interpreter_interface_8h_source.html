200  * Implementation of SouffleProgram interface for an interpreter instance
201  */
202 class InterpreterProgInterface : public SouffleProgram {
203 private:
204  const RamProgram& prog;
205  Interpreter& exec;
206  SymbolTable& symTable;
207  std::vector<InterpreterRelInterface*> interfaces;
208 
209 public:
210  InterpreterProgInterface(Interpreter& interp)
211  : prog(interp.getTranslationUnit().getP()), exec(interp),
212  symTable(interp.getTranslationUnit().getSymbolTable()) {
213  uint32_t id = 0;
214 
215  // Retrieve AST Relations and store them in a map
216  std::map<std::string, const RamRelation*> map;
217  visitDepthFirst(*(prog.getMain()), [&](const RamRelation& rel) { map[rel.getName()] = &rel; });
218 
219  // Build wrapper relations for Souffle's interface
220  for (auto& rel_pair : exec.getRelationMap()) {
221  auto& name = rel_pair.first;
222  auto& interpreterRel = *rel_pair.second;
223  assert(map[name]);
224  const RamRelation& rel = *map[name];
225 
226  // construct types and names vectors
227  std::vector<std::string> types;
228  std::vector<std::string> attrNames;
229  for (size_t i = 0; i < rel.getArity(); i++) {
230  std::string t = rel.getArgTypeQualifier(i);
231  types.push_back(t);
232 
233  std::string n = rel.getArg(i);
234  attrNames.push_back(n);
235  }
236  InterpreterRelInterface* interface = new InterpreterRelInterface(interpreterRel, symTable,
237  rel.getName(), types, attrNames, rel.isInput(), rel.isOutput(), id);
238  interfaces.push_back(interface);
239  addRelation(rel.getName(), interface, rel.isInput(), rel.isOutput());
240  id++;
241  }
242  }
243  ~InterpreterProgInterface() override {
244  for (auto* interface : interfaces) {
245  delete interface;
246  }
247  }
