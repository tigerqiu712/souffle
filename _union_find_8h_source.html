97  /**
98  * Equivalent to the find() function in union/find
99  * Find the highest ancestor of the provided node - flattening as we go
100  * @param x the node to find the parent of, whilst flattening its set-tree
101  * @return The parent of x
102  */
103  parent_t findNode(parent_t x, bool isStrong = true) {
104  // If x is its own parent return immediately
105  if (x == b2p(get(x))) {
106  return x;
107  }
108 
109  isStale.store(true);
110  mapStale.store(true);
111 
112  block_t xState = get(x);
113  parent_t newParent = findNode(b2p(xState));
114  block_t newState = pr2b(newParent, b2r(xState));
115 
116  if (isStrong)
117  this->get(x).compare_exchange_strong(xState, newState);
118  else
119  this->get(x).compare_exchange_weak(xState, newState);
120 
121  return newParent;
122  }
123 
124  /**
125  * Read only version of findNode
126  * i.e. it doesn't compress the tree when searching - it only finds the top representative
127  * @param x the node to find the rep of
128  * @return the representative that is found
129  */
130  parent_t readOnlyFindNode(parent_t x) const {
131  block_t ii = get(x);
132  parent_t p = b2p(ii);
133  if (x == p) return x;
134  return readOnlyFindNode(p);

297  /**
298  * Check whether the two indices are in the same set
299  * @param x node to be checked
300  * @param y node to be checked

303  bool sameSet(parent_t x, parent_t y) {

313  /**
314  * Union the two specified index nodes
315  * @param x node to be unioned
316  * @param y node to be unioned
317  */
318  void unionNodes(parent_t x, parent_t y) {
319  while (true) {
320  x = findNode(x, false);
321  y = findNode(y, false);
322 
323  // no need to union if both already in same set
324  if (x == y) return;
325 
326  isStale.store(true);
327  mapStale.store(true);
328 
329  rank_t xrank = b2r(get(x));
330  rank_t yrank = b2r(get(y));
331 
332  // if x comes before y (better rank or earlier & equal node)
333  if (xrank > yrank || ((xrank == yrank) && x > y)) {
334  std::swap(x, y);
335  std::swap(xrank, yrank);
336  }
337  // join the trees together
338  // perhaps we can optimise the use of compare_exchange_strong here, as we're in a pessimistic loop
339  if (!updateRoot(x, xrank, y, yrank, true)) continue;
340  if (xrank == yrank) updateRoot(y, yrank, y, yrank + 1, false);
341  break;
342  }
343  }

369  auto xpar = (parent_t)a_blocks.size();

384  /**

386  * Keys of the map are the representatives
387  * Values are the representative's children

389  void genMap() {

404  repToSubords[b2p(this->get(i))].add(i);

451 template <typename SparseDomain>
452 class SparseDisjointSet {
453  // lock on write operations - STL containers only support 1 writer at all times
454  // mutable std::mutex modLock;
455  // read/write lock on sparseToDenseMap & denseToSparseMap
456  mutable souffle::shared_mutex mapsLock;
457 
458  DisjointSet ds;
459 
460  // values stored in here to those in the dense disjoint set
461  std::unordered_map<SparseDomain, parent_t> sparseToDenseMap;
462  std::vector<SparseDomain> denseToSparseMap;
463 
464 private:
465  /**
466  * Retrieve dense encoding, adding it in if non-existent
467  * @param in the sparse value
468  * @return the corresponding dense value
469  */
470  parent_t toDense(const SparseDomain in) {
471  mapsLock.lock_shared();
472  auto it = sparseToDenseMap.find(in);
473 
474  // use the pre-existing value
475  if (it != sparseToDenseMap.end()) {
476  auto ret = it->second;
477  mapsLock.unlock_shared();
478  return ret;
479  }
480  mapsLock.unlock_shared();
481 
482  // create node in conversion
483  mapsLock.lock();
484 
485  it = sparseToDenseMap.find(in);
486  // use the pre-existing value (as perhaps, it may have been written twice)
487  if (it != sparseToDenseMap.end()) {
488  auto ret = it->second;
489  mapsLock.unlock();
490  return ret;
491  }
492 
493  size_t j = denseToSparseMap.size();
494  // check if we create a dense value outside of the bounds that can be stored
495  if (j > std::numeric_limits<parent_t>::max()) throw std::runtime_error("out of bounds dense value");
496  auto jClipped = static_cast<parent_t>(j);
497 
498  // we create the node
499  ds.makeNode();
500  denseToSparseMap.push_back(in);
501  sparseToDenseMap[in] = jClipped;
502 
503  mapsLock.unlock();
504 
505  return jClipped;
506  }
507 
508 public:
509  // warning! not thread safe, do not perform copy operations
510  SparseDisjointSet& operator=(const SparseDisjointSet& old) {
511  if (&old == this) return *this;
512 
513  ds = old.ds;
514  sparseToDenseMap = old.sparseToDenseMap;
515  denseToSparseMap = old.denseToSparseMap;
516 
517  return *this;
518  }
