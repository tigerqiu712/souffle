102 void ParserDriver::addRelation(std::unique_ptr<AstRelation> r) {
103  const auto& name = r->getName();
104  if (AstRelation* prev = translationUnit->getProgram()->getRelation(name)) {
105  Diagnostic err(Diagnostic::ERROR,
106  DiagnosticMessage("Redefinition of relation " + toString(name), r->getSrcLoc()),
107  {DiagnosticMessage("Previous definition", prev->getSrcLoc())});
108  translationUnit->getErrorReport().addDiagnostic(err);
109  } else {
110  if (r->isInput()) {
111  translationUnit->getErrorReport().addWarning(
112  "Deprecated input qualifier was used in relation " + toString(name), r->getSrcLoc());
113  }
114  if (r->isOutput()) {
115  translationUnit->getErrorReport().addWarning(
116  "Deprecated output qualifier was used in relation " + toString(name), r->getSrcLoc());
117  }
118  if (r->isPrintSize()) {
119  translationUnit->getErrorReport().addWarning(
120  "Deprecated printsize qualifier was used in relation " + toString(name), r->getSrcLoc());
121  }
