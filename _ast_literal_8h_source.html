57 /**
58  * Subclass of Literal that represents the use of a relation
59  * either in the head or in the body of a Clause, e.g., parent(x,y).
60  * The arguments of the atom can be variables or constants.

62 class AstAtom : public AstLiteral {
63 protected:
64  /** Name of the atom */

66

86  void setName(const AstRelationIdentifier& n) {
87  name = n;
88  }

91  const AstAtom* getAtom() const override {
92  return this;
93  }

96  void addArgument(std::unique_ptr<AstArgument> arg) {
97  arguments.push_back(std::move(arg));
98  }
99 
100  /** Return the i-th argument of the atom */
101  AstArgument* getArgument(size_t idx) const {
102  return arguments[idx].get();
103  }
104 
105  /** Replace the argument at the given index with the given argument */
106  void setArgument(size_t idx, std::unique_ptr<AstArgument> newArg) {
107  arguments[idx].swap(newArg);
108  }
109 
110  /** Provides access to the list of arguments of this atom */
111  std::vector<AstArgument*> getArguments() const {
112  return toPtrVector(arguments);
113  }
114 
115  /** Return the number of arguments */
116  size_t argSize() const {
117  return arguments.size();
118  }
119 
120  /** Output to a given stream */
121  void print(std::ostream& os) const override {
122  os << getName() << "(";
123 
124  for (size_t i = 0; i < arguments.size(); ++i) {
125  if (i != 0) {
126  os << ",";
127  }
128  if (arguments[i] != nullptr) {
129  arguments[i]->print(os);
130  } else {
131  os << "_";
132  }
133  }
134  os << ")";
135  }
136 
137  /** Creates a clone of this AST sub-structure */
138  AstAtom* clone() const override {
139  auto res = new AstAtom(name);
140  res->setSrcLoc(getSrcLoc());
141  for (const auto& cur : arguments) {
142  res->arguments.push_back(std::unique_ptr<AstArgument>(cur->clone()));
143  }
144  return res;
145  }
146 
147  /** Mutates this node */
148  void apply(const AstNodeMapper& map) override {
149  for (auto& arg : arguments) {
150  arg = map(std::move(arg));
151  }
152  }
153 
154  /** Obtains a list of all embedded child nodes */
155  std::vector<const AstNode*> getChildNodes() const override {
156  std::vector<const AstNode*> res;
157  for (auto& cur : arguments) {
158  res.push_back(cur.get());
159  }
160  return res;
161  }
162 
163 protected:
164  /** Implements the node comparison for this node type */
165  bool equal(const AstNode& node) const override {
166  assert(nullptr != dynamic_cast<const AstAtom*>(&node));
167  const auto& other = static_cast<const AstAtom&>(node);
168  return name == other.name && equal_targets(arguments, other.arguments);
169  }
