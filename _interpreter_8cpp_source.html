475  void visitAggregate(const RamAggregate& aggregate) override {
476  // get the targeted relation
477  const InterpreterRelation& rel = interpreter.getRelation(aggregate.getRelation());
478 
479  // initialize result
480  RamDomain res = 0;
481  switch (aggregate.getFunction()) {
482  case RamAggregate::MIN:
483  res = MAX_RAM_DOMAIN;
484  break;
485  case RamAggregate::MAX:
486  res = MIN_RAM_DOMAIN;
487  break;
488  case RamAggregate::COUNT:
489  res = 0;
490  break;
491  case RamAggregate::SUM:
492  res = 0;
493  break;
494  }
495 
496  // init temporary tuple for this level
497  auto arity = rel.getArity();
498 
499  // get lower and upper boundaries for iteration
500  const auto& pattern = aggregate.getPattern();
501  RamDomain low[arity];
502  RamDomain hig[arity];
503 
504  for (size_t i = 0; i < arity; i++) {
505  if (pattern[i] != nullptr) {
506  low[i] = interpreter.evalVal(*pattern[i], ctxt);
507  hig[i] = low[i];
508  } else {
509  low[i] = MIN_RAM_DOMAIN;
510  hig[i] = MAX_RAM_DOMAIN;
511  }
512  }
513 
514  // obtain index
515  auto idx = rel.getIndex(aggregate.getRangeQueryColumns());
516 
517  // get iterator range
518  auto range = idx->lowerUpperBound(low, hig);
519 
520  // check for emptiness
521  if (aggregate.getFunction() != RamAggregate::COUNT) {
522  if (range.first == range.second) {
523  return; // no elements => no min/max
524  }
525  }
526 
527  // iterate through values
528  for (auto ip = range.first; ip != range.second; ++ip) {
529  // link tuple
530  const RamDomain* data = *(ip);
531  ctxt[aggregate.getLevel()] = data;
532 
533  // count is easy
534  if (aggregate.getFunction() == RamAggregate::COUNT) {
535  res++;
536  continue;
537  }
538 
539  // aggregation is a bit more difficult
540 
541  // eval target expression
542  RamDomain cur = interpreter.evalVal(*aggregate.getTargetExpression(), ctxt);
543 
544  switch (aggregate.getFunction()) {
545  case RamAggregate::MIN:
546  res = std::min(res, cur);
547  break;
548  case RamAggregate::MAX:
549  res = std::max(res, cur);
550  break;
551  case RamAggregate::COUNT:
552  res = 0;
553  break;
554  case RamAggregate::SUM:
555  res += cur;
556  break;
557  }
558  }
559

561  RamDomain tuple[1];

563  ctxt[aggregate.getLevel()] = tuple;

566  auto condition = aggregate.getCondition();

826 /** Execute main program of a translation unit */
827 void Interpreter::executeMain() {
828  SignalHandler::instance()->set();
829  if (Global::config().has("verbose")) {
830  SignalHandler::instance()->enableLogging();
831  }
832  const RamStatement& main = *translationUnit.getP().getMain();
833 
834  if (!Global::config().has("profile")) {
835  evalStmt(main);
836  } else {
837  ProfileEventSingleton::instance().setOutputFile(Global::config().get("profile"));
838  // Prepare the frequency table for threaded use
839  visitDepthFirst(main, [&](const RamSearch& node) {
840  if (!node.getProfileText().empty()) {
841  frequencies.emplace(node.getProfileText(), std::map<size_t, size_t>());
842  }
843  });
844  // Enable profiling for execution of main
845  ProfileEventSingleton::instance().startTimer();
846  ProfileEventSingleton::instance().makeTimeEvent("@time;starttime");
847  // Store configuration
848  for (const auto& cur : Global::config().data()) {
849  ProfileEventSingleton::instance().makeConfigRecord(cur.first, cur.second);
850  }
851  // Store count of relations
852  size_t relationCount = 0;
853  visitDepthFirst(main, [&](const RamCreate& create) {
854  if (create.getRelation().getName()[0] != '@') ++relationCount;
855  });
856  ProfileEventSingleton::instance().makeConfigRecord("relationCount", std::to_string(relationCount));
857 
858  // Store count of rules
859  size_t ruleCount = 0;
860  visitDepthFirst(main, [&](const RamInsert& rule) { ++ruleCount; });
861  ProfileEventSingleton::instance().makeConfigRecord("ruleCount", std::to_string(ruleCount));
862 
863  evalStmt(main);
864  ProfileEventSingleton::instance().stopTimer();
865  for (auto const& cur : frequencies) {
866  for (auto const& iter : cur.second) {
867  ProfileEventSingleton::instance().makeQuantityEvent(cur.first, iter.second, iter.first);
868  }
869  }
870  }
871  SignalHandler::instance()->reset();
872 }
