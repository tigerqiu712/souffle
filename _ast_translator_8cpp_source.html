432 std::unique_ptr<RamValue> translateValue(const AstArgument* arg, const ValueIndex& index = ValueIndex()) {
433  std::unique_ptr<RamValue> val;
434  if (!arg) {
435  return val;
436  }
437 
438  if (const auto* var = dynamic_cast<const AstVariable*>(arg)) {
439  assert(index.isDefined(*var) && "variable not grounded");
440  const Location& loc = index.getDefinitionPoint(*var);
441  val = std::make_unique<RamElementAccess>(loc.level, loc.component, loc.name);
442  } else if (dynamic_cast<const AstUnnamedVariable*>(arg)) {
443  return nullptr; // utilised to identify _ values
444  } else if (const auto* c = dynamic_cast<const AstConstant*>(arg)) {
445  val = std::make_unique<RamNumber>(c->getIndex());
446  } else if (const auto* uf = dynamic_cast<const AstUnaryFunctor*>(arg)) {
447  val = std::make_unique<RamUnaryOperator>(uf->getFunction(), translateValue(uf->getOperand(), index));
448  } else if (const auto* bf = dynamic_cast<const AstBinaryFunctor*>(arg)) {
449  val = std::make_unique<RamBinaryOperator>(
450  bf->getFunction(), translateValue(bf->getLHS(), index), translateValue(bf->getRHS(), index));
451  } else if (const auto* tf = dynamic_cast<const AstTernaryFunctor*>(arg)) {
452  val = std::make_unique<RamTernaryOperator>(tf->getFunction(), translateValue(tf->getArg(0), index),
453  translateValue(tf->getArg(1), index), translateValue(tf->getArg(2), index));
454  } else if (dynamic_cast<const AstCounter*>(arg)) {
455  val = std::make_unique<RamAutoIncrement>();
456  } else if (const auto* init = dynamic_cast<const AstRecordInit*>(arg)) {
457  std::vector<std::unique_ptr<RamValue>> values;
458  for (const auto& cur : init->getArguments()) {
459  values.push_back(translateValue(cur, index));
460  }
461  val = std::make_unique<RamPack>(std::move(values));
462  } else if (const auto* agg = dynamic_cast<const AstAggregator*>(arg)) {
463  // here we look up the location the aggregation result gets bound
464  auto loc = index.getAggregatorLocation(*agg);
465  val = std::make_unique<RamElementAccess>(loc.level, loc.component, loc.name);
466  } else if (const auto* subArg = dynamic_cast<const AstSubroutineArgument*>(arg)) {
467  val = std::make_unique<RamArgument>(subArg->getNumber());
468  } else {
469  std::cout << "Unsupported node type of " << arg << ": " << typeid(*arg).name() << "\n";
470  assert(false && "unknown AST node type not permissible");
471  }
472 
473  return val;
474 }

482 std::unique_ptr<RamStatement> AstTranslator::translateClause(const AstClause& clause,
483  const AstProgram* program, const TypeEnvironment* typeEnv, const AstClause& originalClause,
484  int version, bool ret, bool hashset) {
485  // check whether there is an imposed order constraint
486  if (clause.getExecutionPlan() && clause.getExecutionPlan()->hasOrderFor(version)) {
487  // get the imposed order
488  const AstExecutionOrder& order = clause.getExecutionPlan()->getOrderFor(version);
489 
490  // create a copy and fix order
491  std::unique_ptr<AstClause> copy(clause.clone());
492 
493  // Change order to start at zero
494  std::vector<unsigned int> newOrder(order.size());
495  std::transform(order.begin(), order.end(), newOrder.begin(),
496  [](unsigned int i) -> unsigned int { return i - 1; });
497 
498  // re-order atoms
499  copy->reorderAtoms(newOrder);
500 
501  // clear other order and fix plan
502  copy->clearExecutionPlan();
503  copy->setFixedExecutionPlan();
504 
505  // translate reordered clause
506  return translateClause(*copy, program, typeEnv, originalClause, version, false, hashset);
507  }
508 
509  // get extract some details
510  const AstAtom& head = *clause.getHead();
511 
512  // a utility to translate atoms to relations
513  auto getRelation = [&](const AstAtom* atom) {
514  return getRamRelation((program ? getAtomRelation(atom, program) : nullptr), typeEnv,
515  getRelationName(atom->getName()), atom->getArity(), false, hashset);
516  };
517 
518  // handle facts
519  if (clause.isFact()) {
520  // translate arguments
521  std::vector<std::unique_ptr<RamValue>> values;
522  for (auto& arg : clause.getHead()->getArguments()) {
523  values.push_back(translateValue(*arg));
524  }
525 
526  // create a fact statement
527  return std::make_unique<RamFact>(getRelation(&head), std::move(values));
528  }
529 
530  // the rest should be rules
531  assert(clause.isRule());
532 
533  // -- index values in rule --
534 
535  // create value index
536  ValueIndex valueIndex;
537 
538  // the order of processed operations
539  std::vector<const AstNode*> op_nesting;
540 
541  int level = 0;
542  for (AstAtom* atom : clause.getAtoms()) {
543  // index nested variables and records
544  using arg_list = std::vector<AstArgument*>;
545  // std::map<const arg_list*, int> arg_level;
546  std::map<const AstNode*, std::unique_ptr<arg_list>> nodeArgs;
547 
548  std::function<arg_list*(const AstNode*)> getArgList = [&](const AstNode* curNode) {
549  if (!nodeArgs.count(curNode)) {
550  if (auto rec = dynamic_cast<const AstRecordInit*>(curNode)) {
551  nodeArgs.insert(std::make_pair(curNode, std::make_unique<arg_list>(rec->getArguments())));
552  } else if (auto atom = dynamic_cast<const AstAtom*>(curNode)) {
553  nodeArgs.insert(
554  std::make_pair(curNode, std::make_unique<arg_list>(atom->getArguments())));
555  } else {
556  assert(false && "node type doesn't have arguments!");
557  }
558  }
559  arg_list* cur = nodeArgs[curNode].get();
560  return cur;
561  };
562 
563  std::map<const arg_list*, int> arg_level;
564  nodeArgs.insert(std::make_pair(atom, std::make_unique<arg_list>(atom->getArguments())));
565  // the atom is obtained at the current level
566  arg_level[nodeArgs[atom].get()] = level;
567  op_nesting.push_back(atom);
568 
569  // increment nesting level for the atom
570  level++;
571 
572  // relation
573  std::unique_ptr<RamRelation> relation = getRelation(atom);
574 
575  std::function<void(const AstNode*)> indexValues = [&](const AstNode* curNode) {
576  arg_list* cur = getArgList(curNode);
577  for (size_t pos = 0; pos < cur->size(); pos++) {
578  // get argument
579  auto& arg = (*cur)[pos];
580 
581  // check for variable references
582  if (auto var = dynamic_cast<const AstVariable*>(arg)) {
583  if (pos < relation->getArity()) {
584  valueIndex.addVarReference(*var, arg_level[cur], pos, relation->getArg(pos));
585  } else {
586  valueIndex.addVarReference(*var, arg_level[cur], pos);
587  }
588  }
589 
590  // check for nested records
591  if (auto rec = dynamic_cast<const AstRecordInit*>(arg)) {
592  // introduce new nesting level for unpack
593  int unpack_level = level++;
594  op_nesting.push_back(rec);
595  arg_level[getArgList(rec)] = unpack_level;
596  valueIndex.setRecordUnpackLevel(*rec, unpack_level);
597 
598  // register location of record
599  valueIndex.setRecordDefinition(*rec, arg_level[cur], pos);
600 
601  // resolve nested components
602  indexValues(rec);
603  }
604  }
605  };
606 
607  indexValues(atom);
608  }
609 
610  // add aggregation functions
611  std::vector<const AstAggregator*> aggregators;
612  visitDepthFirstPostOrder(clause, [&](const AstAggregator& cur) {
613  // add each aggregator expression only once
614  if (any_of(aggregators, [&](const AstAggregator* agg) { return *agg == cur; })) {
615  return;
616  }
617 
618  int aggLoc = level++;
619  valueIndex.setAggregatorLocation(cur, Location({aggLoc, 0}));
620 
621  // bind aggregator variables to locations
622  assert(nullptr != dynamic_cast<const AstAtom*>(cur.getBodyLiterals()[0]));
623  const AstAtom& atom = static_cast<const AstAtom&>(*cur.getBodyLiterals()[0]);
624  for (size_t pos = 0; pos < atom.getArguments().size(); ++pos) {
625  if (const auto* var = dynamic_cast<const AstVariable*>(atom.getArgument(pos))) {
626  valueIndex.addVarReference(*var, aggLoc, (int)pos, getRelation(&atom)->getArg(pos));
627  }
628  };
629 
630  // and remember aggregator
631  aggregators.push_back(&cur);
632  });
633 
634  // -- create RAM statement --
635 
636  // begin with projection
637  std::unique_ptr<RamOperation> op;
638  if (ret) {
639  auto* returnValue = new RamReturn(level);
640 
641  // get all values in the body
642  for (AstLiteral* lit : clause.getBodyLiterals()) {
643  if (auto atom = dynamic_cast<AstAtom*>(lit)) {
644  for (AstArgument* arg : atom->getArguments()) {
645  returnValue->addValue(translateValue(arg, valueIndex));
646  }
647  } else if (auto neg = dynamic_cast<AstNegation*>(lit)) {
648  for (AstArgument* arg : neg->getAtom()->getArguments()) {
649  returnValue->addValue(translateValue(arg, valueIndex));
650  }
651  } else if (auto neg = dynamic_cast<AstProvenanceNegation*>(lit)) {
652  for (size_t i = 0; i < neg->getAtom()->getArguments().size() - 2; i++) {
653  auto arg = neg->getAtom()->getArguments()[i];
654  returnValue->addValue(translateValue(arg, valueIndex));
655  }
656  returnValue->addValue(std::make_unique<RamNumber>(-1));
657  returnValue->addValue(std::make_unique<RamNumber>(-1));
658  }
659  }
660 
661  op = std::unique_ptr<RamOperation>(returnValue);
662  } else {
663  std::unique_ptr<RamProject> project = std::make_unique<RamProject>(getRelation(&head), level);
664 
665  for (AstArgument* arg : head.getArguments()) {
666  project->addArg(translateValue(arg, valueIndex));
667  }
668 
669  // check existence for original tuple if we have provenance
670  // only if we don't compile
671  if (Global::config().has("provenance") &&
672  ((!Global::config().has("compile") && !Global::config().has("dl-program") &&
673  !Global::config().has("generate")))) {
674  auto uniquenessEnforcement = std::make_unique<RamNotExists>(getRelation(&head));
675  auto arity = head.getArity() - 2;
676 
677  bool isVolatile = true;
678  // add args for original tuple
679  for (size_t i = 0; i < arity; i++) {
680  auto arg = head.getArgument(i);
681 
682  // don't add counters
683  visitDepthFirst(*arg, [&](const AstCounter& cur) { isVolatile = false; });
684  uniquenessEnforcement->addArg(translateValue(arg, valueIndex));
685  }
686 
687  // add two unnamed args for provenance columns
688  uniquenessEnforcement->addArg(nullptr);
689  uniquenessEnforcement->addArg(nullptr);
690 
691  if (isVolatile) {
692  project->addCondition(std::move(uniquenessEnforcement), *project);
693  }
694  }
695 
696  // build up insertion call
697  op = std::move(project); // start with innermost
698  }
699 
700  // add aggregator levels
701  for (auto it = aggregators.rbegin(); it != aggregators.rend(); ++it) {
702  const AstAggregator* cur = *it;
703  level--;
704 
705  // translate aggregation function
706  RamAggregate::Function fun = RamAggregate::MIN;
707  switch (cur->getOperator()) {
708  case AstAggregator::min:
709  fun = RamAggregate::MIN;
710  break;
711  case AstAggregator::max:
712  fun = RamAggregate::MAX;
713  break;
714  case AstAggregator::count:
715  fun = RamAggregate::COUNT;
716  break;
717  case AstAggregator::sum:
718  fun = RamAggregate::SUM;
719  break;
720  }
721 
722  // translate target expression
723  std::unique_ptr<RamValue> value = translateValue(cur->getTargetExpression(), valueIndex);
724 
725  // translate body literal
726  assert(cur->getBodyLiterals().size() == 1 && "Unsupported complex aggregation body encountered!");
727  const AstAtom* atom = dynamic_cast<const AstAtom*>(cur->getBodyLiterals()[0]);
728  assert(atom && "Unsupported complex aggregation body encountered!");
729 
730  // add Ram-Aggregation layer
731  op = std::make_unique<RamAggregate>(std::move(op), fun, std::move(value), getRelation(atom));
732 
733  // add constant constraints
734  for (size_t pos = 0; pos < atom->argSize(); ++pos) {
735  if (auto* c = dynamic_cast<AstConstant*>(atom->getArgument(pos))) {
736  op->addCondition(std::make_unique<RamBinaryRelation>(BinaryConstraintOp::EQ,
737  std::make_unique<RamElementAccess>(level, pos, getRelation(atom)->getArg(pos)),
738  std::make_unique<RamNumber>(c->getIndex())));
739  }
740  }
741  }
742 
743  // build operation bottom-up
744  while (!op_nesting.empty()) {
745  // get next operator
746  const AstNode* cur = op_nesting.back();
747  op_nesting.pop_back();
748 
749  // get current nesting level
750  auto level = op_nesting.size();
751 
752  if (const auto* atom = dynamic_cast<const AstAtom*>(cur)) {
753  // find out whether a "search" or "if" should be issued
754  bool isExistCheck = !valueIndex.isSomethingDefinedOn(level);
755  for (size_t pos = 0; pos < atom->argSize(); ++pos) {
756  if (dynamic_cast<AstAggregator*>(atom->getArgument(pos))) {
757  isExistCheck = false;
758  }
759  }
760 
761  // add a scan level
762  if (Global::config().has("profile")) {
763  std::stringstream ss;
764  ss << clause.getHead()->getName();
765  std::string relName = ss.str();
766  ss.str("");
767 
768  if (modifiedIdMap.find(relName) != modifiedIdMap.end()) {
769  relName = modifiedIdMap[relName];
770  }
771 
772  ss << "@frequency-atom" << ';';
773  ss << relName << ';';
774  ss << version << ';';
775  ss << stringify(toString(clause)) << ';';
776  ss << stringify(toString(*atom)) << ';';
777  ss << stringify(toString(originalClause)) << ';';
778  ss << level << ';';
779  op = std::make_unique<RamScan>(getRelation(atom), std::move(op), isExistCheck, ss.str());
780  } else {
781  op = std::make_unique<RamScan>(getRelation(atom), std::move(op), isExistCheck);
782  }
783 
784  // add constraints
785  for (size_t pos = 0; pos < atom->argSize(); ++pos) {
786  if (auto* c = dynamic_cast<AstConstant*>(atom->getArgument(pos))) {
787  op->addCondition(std::make_unique<RamBinaryRelation>(BinaryConstraintOp::EQ,
788  std::make_unique<RamElementAccess>(level, pos, getRelation(atom)->getArg(pos)),
789  std::make_unique<RamNumber>(c->getIndex())));
790  } else if (auto* agg = dynamic_cast<AstAggregator*>(atom->getArgument(pos))) {
791  auto loc = valueIndex.getAggregatorLocation(*agg);
792  op->addCondition(std::make_unique<RamBinaryRelation>(BinaryConstraintOp::EQ,
793  std::make_unique<RamElementAccess>(level, pos, getRelation(atom)->getArg(pos)),
794  std::make_unique<RamElementAccess>(loc.level, loc.component, loc.name)));
795  }
796  }
797 
798  // TODO: support constants in nested records!
799  } else if (const auto* rec = dynamic_cast<const AstRecordInit*>(cur)) {
800  // add an unpack level
801  const Location& loc = valueIndex.getDefinitionPoint(*rec);
802  op = std::make_unique<RamLookup>(
803  std::move(op), loc.level, loc.component, rec->getArguments().size());
804 
805  // add constant constraints
806  for (size_t pos = 0; pos < rec->getArguments().size(); ++pos) {
807  if (AstConstant* c = dynamic_cast<AstConstant*>(rec->getArguments()[pos])) {
808  op->addCondition(std::make_unique<RamBinaryRelation>(BinaryConstraintOp::EQ,
809  std::make_unique<RamElementAccess>(level, pos),
810  std::make_unique<RamNumber>(c->getIndex())));
811  }
812  }
813  } else {
814  std::cout << "Unsupported AST node type: " << typeid(*cur).name() << "\n";
815  assert(false && "Unsupported AST node for creation of scan-level!");
816  }
817  }
818 
819  /* add equivalence constraints imposed by variable binding */
820  for (const auto& cur : valueIndex.getVariableReferences()) {
821  // the first appearance
822  const Location& first = *cur.second.begin();
823  // all other appearances
824  for (const Location& loc : cur.second) {
825  if (first != loc) {
826  op->addCondition(std::make_unique<RamBinaryRelation>(BinaryConstraintOp::EQ,
827  std::make_unique<RamElementAccess>(first.level, first.component, first.name),
828  std::make_unique<RamElementAccess>(loc.level, loc.component, loc.name)));
829  }
830  }
831  }
832 
833  /* add conditions caused by atoms, negations, and binary relations */
834  for (const auto& lit : clause.getBodyLiterals()) {
835  // for atoms
836  if (dynamic_cast<const AstAtom*>(lit)) {
837  // covered already within the scan/lookup generation step
838 
839  // for binary relations
840  } else if (auto binRel = dynamic_cast<const AstBinaryConstraint*>(lit)) {
841  std::unique_ptr<RamValue> valLHS = translateValue(binRel->getLHS(), valueIndex);
842  std::unique_ptr<RamValue> valRHS = translateValue(binRel->getRHS(), valueIndex);
843  op->addCondition(std::make_unique<RamBinaryRelation>(binRel->getOperator(),
844  translateValue(binRel->getLHS(), valueIndex),
845  translateValue(binRel->getRHS(), valueIndex)));
846 
847  // for negations
848  } else if (auto neg = dynamic_cast<const AstNegation*>(lit)) {
849  // get contained atom
850  const AstAtom* atom = neg->getAtom();
851 
852  // create constraint
853  RamNotExists* notExists = new RamNotExists(getRelation(atom));
854 
855  auto arity = atom->getArity();
856 
857  // account for two extra provenance columns
858  if (Global::config().has("provenance")) {
859  arity -= 2;
860  }
861 
862  for (size_t i = 0; i < arity; i++) {
863  const auto& arg = atom->getArgument(i);
864  // for (const auto& arg : atom->getArguments()) {
865  notExists->addArg(translateValue(*arg, valueIndex));
866  }
867 
868  // we don't care about the provenance columns when doing the existence check
869  if (Global::config().has("provenance")) {
870  notExists->addArg(nullptr);
871  notExists->addArg(nullptr);
872  }
873 
874  // add constraint
875  op->addCondition(std::unique_ptr<RamCondition>(notExists));
876 
877  // for provenance negation
878  } else if (auto neg = dynamic_cast<const AstProvenanceNegation*>(lit)) {
879  // get contained atom
880  const AstAtom* atom = neg->getAtom();
881 
882  // create constraint
883  RamProvenanceNotExists* notExists = new RamProvenanceNotExists(getRelation(atom));
884 
885  auto arity = atom->getArity();
886 
887  // account for two extra provenance columns
888  if (Global::config().has("provenance")) {
889  arity -= 2;
890  }
891 
892  for (size_t i = 0; i < arity; i++) {
893  const auto& arg = atom->getArgument(i);
894  // for (const auto& arg : atom->getArguments()) {
895  notExists->addArg(translateValue(*arg, valueIndex));
896  }
897 
898  // we don't care about the provenance columns when doing the existence check
899  if (Global::config().has("provenance")) {
900  notExists->addArg(nullptr);
901  // add the height annotation for provenanceNotExists
902  notExists->addArg(translateValue(*(atom->getArgument(arity + 1)), valueIndex));
903  }
904 
905  // add constraint
906  op->addCondition(std::unique_ptr<RamCondition>(notExists));
907  } else {
908  std::cout << "Unsupported node type: " << typeid(*lit).name();
909  assert(false && "Unsupported node type!");
910  }
911  }
912 
913  /* generate the final RAM Insert statement */
914  return std::make_unique<RamInsert>(std::move(op));
915 }

1036 /** generate RAM code for recursive relations in a strongly-connected component */
1037 std::unique_ptr<RamStatement> AstTranslator::translateRecursiveRelation(
1038  const std::set<const AstRelation*>& scc, const AstProgram* program,
1039  const RecursiveClauses* recursiveClauses, const TypeEnvironment& typeEnv) {
1040  // initialize sections
1041  std::unique_ptr<RamStatement> preamble;
1042  std::unique_ptr<RamSequence> updateTable(new RamSequence());
1043  std::unique_ptr<RamStatement> postamble;
1044 
1045  // --- create preamble ---
1046 
1047  // mappings for temporary relations
1048  std::map<const AstRelation*, std::unique_ptr<RamRelation>> rrel;
1049  std::map<const AstRelation*, std::unique_ptr<RamRelation>> relDelta;
1050  std::map<const AstRelation*, std::unique_ptr<RamRelation>> relNew;
1051 
1052  /* Compute non-recursive clauses for relations in scc and push
1053  the results in their delta tables. */
1054  for (const AstRelation* rel : scc) {
1055  std::unique_ptr<RamStatement> updateRelTable;
1056 
1057  /* create two temporary tables for relaxed semi-naive evaluation */
1058  auto relName = getRelationName(rel->getName());
1059  rrel[rel] = getRamRelation(rel, &typeEnv, relName, rel->getArity(), false, rel->isHashset());
1060  relDelta[rel] =
1061  getRamRelation(rel, &typeEnv, "delta_" + relName, rel->getArity(), true, rel->isHashset());
1062  relNew[rel] =
1063  getRamRelation(rel, &typeEnv, "new_" + relName, rel->getArity(), true, rel->isHashset());
1064 
1065  modifiedIdMap[relName] = relName;
1066  modifiedIdMap[relDelta[rel]->getName()] = relName;
1067  modifiedIdMap[relNew[rel]->getName()] = relName;
1068 
1069  /* create update statements for fixpoint (even iteration) */
1070  appendStmt(updateRelTable,
1071  std::make_unique<RamSequence>(
1072  std::make_unique<RamMerge>(std::unique_ptr<RamRelation>(rrel[rel]->clone()),
1073  std::unique_ptr<RamRelation>(relNew[rel]->clone())),
1074  std::make_unique<RamSwap>(std::unique_ptr<RamRelation>(relDelta[rel]->clone()),
1075  std::unique_ptr<RamRelation>(relNew[rel]->clone())),
1076  std::make_unique<RamClear>(std::unique_ptr<RamRelation>(relNew[rel]->clone()))));
1077 
1078  /* measure update time for each relation */
1079  if (Global::config().has("profile")) {
1080  updateRelTable = std::make_unique<RamLogTimer>(std::move(updateRelTable),
1081  LogStatement::cRecursiveRelation(toString(rel->getName()), rel->getSrcLoc()));
1082  }
1083 
1084  /* drop temporary tables after recursion */
1085  appendStmt(postamble,
1086  std::make_unique<RamSequence>(
1087  std::make_unique<RamDrop>(std::unique_ptr<RamRelation>(relDelta[rel]->clone())),
1088  std::make_unique<RamDrop>(std::unique_ptr<RamRelation>(relNew[rel]->clone()))));
1089 
1090  /* Generate code for non-recursive part of relation */
1091  appendStmt(preamble, translateNonRecursiveRelation(*rel, program, recursiveClauses, typeEnv));
1092 
1093  /* Generate merge operation for temp tables */
1094  appendStmt(preamble, std::make_unique<RamMerge>(std::unique_ptr<RamRelation>(relDelta[rel]->clone()),
1095  std::unique_ptr<RamRelation>(rrel[rel]->clone())));
1096 
1097  /* Add update operations of relations to parallel statements */
1098  updateTable->add(std::move(updateRelTable));
1099  }
1100 
1101  // --- build main loop ---
1102 
1103  std::unique_ptr<RamParallel> loopSeq(new RamParallel());
1104 
1105  // create a utility to check SCC membership
1106  auto isInSameSCC = [&](const AstRelation* rel) {
1107  return std::find(scc.begin(), scc.end(), rel) != scc.end();
1108  };
1109 
1110  /* Compute temp for the current tables */
1111  for (const AstRelation* rel : scc) {
1112  std::unique_ptr<RamStatement> loopRelSeq;
1113 
1114  /* Find clauses for relation rel */
1115  for (size_t i = 0; i < rel->clauseSize(); i++) {
1116  AstClause* cl = rel->getClause(i);
1117 
1118  // skip non-recursive clauses
1119  if (!recursiveClauses->recursive(cl)) {
1120  continue;
1121  }
1122 
1123  // each recursive rule results in several operations
1124  int version = 0;
1125  const auto& atoms = cl->getAtoms();
1126  for (size_t j = 0; j < atoms.size(); ++j) {
1127  const AstAtom* atom = atoms[j];
1128  const AstRelation* atomRelation = getAtomRelation(atom, program);
1129 
1130  // only interested in atoms within the same SCC
1131  if (!isInSameSCC(atomRelation)) {
1132  continue;
1133  }
1134 
1135  // modify the processed rule to use relDelta and write to relNew
1136  std::unique_ptr<AstClause> r1(cl->clone());
1137  r1->getHead()->setName(relNew[rel]->getName());
1138  r1->getAtoms()[j]->setName(relDelta[atomRelation]->getName());
1139  if (Global::config().has("provenance")) {
1140  r1->addToBody(std::make_unique<AstProvenanceNegation>(
1141  std::unique_ptr<AstAtom>(cl->getHead()->clone())));
1142  } else {
1143  r1->addToBody(
1144  std::make_unique<AstNegation>(std::unique_ptr<AstAtom>(cl->getHead()->clone())));
1145  }
1146 
1147  // replace wildcards with variables (reduces indices when wildcards are used in recursive
1148  // atoms)
1149  nameUnnamedVariables(r1.get());
1150 
1151  // reduce R to P ...
1152  for (size_t k = j + 1; k < atoms.size(); k++) {
1153  if (isInSameSCC(getAtomRelation(atoms[k], program))) {
1154  AstAtom* cur = r1->getAtoms()[k]->clone();
1155  cur->setName(relDelta[getAtomRelation(atoms[k], program)]->getName());
1156  r1->addToBody(std::make_unique<AstNegation>(std::unique_ptr<AstAtom>(cur)));
1157  }
1158  }
1159 
1160  std::unique_ptr<RamStatement> rule =
1161  translateClause(*r1, program, &typeEnv, *cl, version, false, rel->isHashset());
1162 
1163  /* add logging */
1164  if (Global::config().has("profile")) {
1165  const std::string& relationName = toString(rel->getName());
1166  const SrcLocation& srcLocation = cl->getSrcLoc();
1167  const std::string clauseText = stringify(toString(*cl));
1168  const std::string logTimerStatement =
1169  LogStatement::tRecursiveRule(relationName, version, srcLocation, clauseText);
1170  const std::string logSizeStatement =
1171  LogStatement::nRecursiveRule(relationName, version, srcLocation, clauseText);
1172  rule = std::make_unique<RamSequence>(
1173  std::make_unique<RamLogTimer>(std::move(rule), logTimerStatement),
1174  std::make_unique<RamLogSize>(
1175  std::unique_ptr<RamRelation>(relNew[rel]->clone()), logSizeStatement));
1176  }
1177 
1178  // add debug info
1179  std::ostringstream ds;
1180  ds << toString(*cl) << "\nin file ";
1181  ds << cl->getSrcLoc();
1182  rule = std::make_unique<RamDebugInfo>(std::move(rule), ds.str());
1183 
1184  // add to loop body
1185  appendStmt(loopRelSeq, std::move(rule));
1186 
1187  // increment version counter
1188  version++;
1189  }
1190  assert(cl->getExecutionPlan() == nullptr || version > cl->getExecutionPlan()->getMaxVersion());
1191  }
1192 
1193  // if there was no rule, continue
1194  if (!loopRelSeq) {
1195  continue;
1196  }
1197 
1198  // label all versions
1199  if (Global::config().has("profile")) {
1200  const std::string& relationName = toString(rel->getName());
1201  const SrcLocation& srcLocation = rel->getSrcLoc();
1202  const std::string logTimerStatement = LogStatement::tRecursiveRelation(relationName, srcLocation);
1203  const std::string logSizeStatement = LogStatement::nRecursiveRelation(relationName, srcLocation);
1204  loopRelSeq = std::make_unique<RamLogTimer>(std::move(loopRelSeq), logTimerStatement);
1205  appendStmt(loopRelSeq,
1206  std::make_unique<RamLogSize>(
1207  std::unique_ptr<RamRelation>(relNew[rel]->clone()), logSizeStatement));
1208  }
1209 
1210  /* add rule computations of a relation to parallel statement */
1211  loopSeq->add(std::move(loopRelSeq));
1212  }
1213 
1214  /* construct exit conditions for odd and even iteration */
1215  auto addCondition = [](std::unique_ptr<RamCondition>& cond, std::unique_ptr<RamCondition> clause) {
1216  cond = ((cond) ? std::make_unique<RamAnd>(std::move(cond), std::move(clause)) : std::move(clause));
1217  };
1218 
1219  std::unique_ptr<RamCondition> exitCond;
1220  for (const AstRelation* rel : scc) {
1221  addCondition(
1222  exitCond, std::make_unique<RamEmpty>(std::unique_ptr<RamRelation>(relNew[rel]->clone())));
1223  }
1224 
1225  /* construct fixpoint loop */
1226  std::unique_ptr<RamStatement> res;
1227  if (preamble) appendStmt(res, std::move(preamble));
1228  if (!loopSeq->getStatements().empty() && exitCond && updateTable) {
