50  * Read and return the next tuple.
51  *
52  * Returns nullptr if no tuple was readable.

55  std::unique_ptr<RamDomain[]> readNextTuple() override {
56  if (sqlite3_step(selectStatement) != SQLITE_ROW) {
57  return nullptr;
58  }
59 
60  std::unique_ptr<RamDomain[]> tuple = std::make_unique<RamDomain[]>(symbolMask.getArity());
61 
62  uint32_t column;
63  for (column = 0; column < symbolMask.getArity(); column++) {
64  std::string element(reinterpret_cast<const char*>(sqlite3_column_text(selectStatement, column)));
65 
66  if (element.empty()) {
67  element = "n/a";
68  }
69  if (symbolMask.isSymbol(column)) {
70  tuple[column] = symbolTable.unsafeLookup(element);
71  } else {
72  try {
73 #if RAM_DOMAIN_SIZE == 64
74  tuple[column] = std::stoll(element);
75 #else
76  tuple[column] = std::stoi(element);
77 #endif
78  } catch (...) {
79  std::stringstream errorMessage;
80  errorMessage << "Error converting number in column " << (column) + 1;
81  throw std::invalid_argument(errorMessage.str());
82  }
83  }
84  }
85 
86  if (isProvenance) {
87  tuple[symbolMask.getArity() - 2] = 0;
88  tuple[symbolMask.getArity() - 1] = 0;
89  }
90 
91  return tuple;
92  }

103  error << "SQL error: " << errorMessage << "\n";

134  void checkTableExists() {
135  sqlite3_stmt* tableStatement;
136  std::stringstream selectSQL;
137  selectSQL << "SELECT count(*) FROM sqlite_master WHERE type IN ('table', 'view') AND ";
138  selectSQL << " name IN ('" << relationName << "', '_" << relationName << "');";
139  const char* tail = nullptr;
140 
141  if (sqlite3_prepare_v2(db, selectSQL.str().c_str(), -1, &tableStatement, &tail) != SQLITE_OK) {
142  throwError("SQLite error in sqlite3_prepare_v2: ");
143  }
144 
145  if (sqlite3_step(tableStatement) == SQLITE_ROW) {
146  int count = sqlite3_column_int(tableStatement, 0);
147  if (count == 2) {
148  sqlite3_finalize(tableStatement);
149  return;
150  }
151  }
152  sqlite3_finalize(tableStatement);
153  throw std::invalid_argument("Required table and view does not exist for relation " + relationName);
154  }
155  const std::string& dbFilename;
156  const std::string& relationName;
157  sqlite3_stmt* selectStatement = nullptr;
